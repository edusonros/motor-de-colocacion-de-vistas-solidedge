        Dim objApplication As SolidEdgeFramework.Application = Nothing
        Dim objDraftDoc As SolidEdgeDraft.DraftDocument = Nothing
        Dim objType As Type = Nothing
        Dim objDimension As SolidEdgeFrameworkSupport.Dimension = Nothing
        Dim objDimensions As SolidEdgeFrameworkSupport.Dimensions = Nothing
        Dim Sheet As SolidEdgeDraft.Sheet = Nothing

        Try

            Try
                objApplication = Marshal.GetActiveObject("SolidEdge.Application")
            Catch ex As Exception
                ' Get the type from the Solid Edge ProgID
                objType = Type.GetTypeFromProgID("SolidEdge.Application")
                ' Start Solid Edge
                objApplication = Activator.CreateInstance(objType)
            Finally
                ' Make Solid Edge visible
                objApplication.Visible = True
            End Try


            ' open the document mentioned in the doc files.

            objDraftDoc = objApplication.ActiveDocument
            If objDraftDoc Is Nothing Then
                MsgBox("Could not open document ")
            End If

            Sheet = objDraftDoc.ActiveSheet

            If (Sheet Is Nothing) Then
                MsgBox("No active sheet in the document")

            End If

            objDimensions = Sheet.Dimensions
            objDimension = objDimensions.Item(1)

            '****************DIMENSIONS*****************
            '***************unit type*******************
            objDimension.UnitsType = 1

            Dim dimUnitType As Integer

            dimUnitType = objDimension.UnitsType

            If dimUnitType <> 1 Then
                MsgBox("Failed to get/set unit type")

            End If

            '***************Discrete values*******************
            Dim inputDiscreteValues(0 To 3) As Double
            Dim outputDiscreteValues(0) As Double

            inputDiscreteValues(0) = 0.0255
            inputDiscreteValues(1) = 0.26
            inputDiscreteValues(2) = 0.265
            inputDiscreteValues(3) = 0.27

            objDimension.SetValueDiscreteValues(inputDiscreteValues)

            objDimension.GetValueDiscreteValues(outputDiscreteValues)

            If outputDiscreteValues(0) <> inputDiscreteValues(0) Or
                outputDiscreteValues(1) <> inputDiscreteValues(1) Or
                outputDiscreteValues(2) <> inputDiscreteValues(2) Or
                outputDiscreteValues(3) <> inputDiscreteValues(3) Then
                MsgBox("Failed to get/set discrete values")

            End If

            '***************Add discrete value*******************
            objDimension.AddDiscreteValue(0.22)

            objDimension.GetValueDiscreteValues(outputDiscreteValues)

            If outputDiscreteValues(2) <> 0.22 Then
                MsgBox("Failed to add discrete values")
            End If

            '***************Remove discrete value*******************
            objDimension.RemoveDiscreteValue(0.265)

            objDimension.GetValueDiscreteValues(outputDiscreteValues)

            If outputDiscreteValues(4) <> 0.27 Then
                MsgBox("Failed to remove discrete values")
            End If

            '***************Range value*******************
            objDimension.SetValueRangeValues(0.01, 5, 1)

            Dim LowValue As Double
            Dim HighValue As Double
            Dim condition As Integer

            objDimension.GetValueRangeValues(LowValue, condition, HighValue)

            If LowValue <> 0.01 Or
                condition <> 5 Or
                HighValue <> 1 Then
                MsgBox("Failed to set/get range values")
            End If

            '***************Range value low*******************
            objDimension.SetValueRangeLowValue(0.001)

            objDimension.GetValueRangeValues(LowValue, condition, HighValue)

            If LowValue <> 0.001 Then
                MsgBox("Failed to set/get low range values")

            End If

            '***************Range value high*******************
            objDimension.SetValueRangeHighValue(2)

            objDimension.GetValueRangeValues(LowValue, condition, HighValue)

            If HighValue <> 2 Then
                MsgBox("Failed to set/get high range values")

            End If

            '*******************Get out of range value****************
            Dim dBadValue As Double
            Dim dimensionValue As Double = 3.0

            'objDimension.Value = dimensionValue
            dBadValue = objDimension.GetValueOutOfRange()

            If dBadValue <> 0.02 Then
                MsgBox("Failed to get out of range value")

            End If

            '*******************Clear limits or discrete values****************
            Dim count As Integer
            objDimension.ClearLimitsOrDiscreteValues()

            objDimension.GetValueDiscreteValues(outputDiscreteValues)
            outputDiscreteValues.GetLength(count)

            If count <> 0 Then
                MsgBox("Failed to clear limits or discrete values")
            End If

        Catch ex As Exception
            MsgBox(ex.ToString)

        Finally

            ' close the doc
            objDraftDoc.Close(False)
            objDraftDoc = Nothing
            objApplication = Nothing

        End Try


    End Sub

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
public class Form1
{
    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgeDraft.DraftDocument objDraftDoc = null;
        Type objType = null;
        SolidEdgeFrameworkSupport.Dimension objDimension = null;
        SolidEdgeFrameworkSupport.Dimensions objDimensions = null;
        SolidEdgeDraft.Sheet Sheet = null;

        try
        {

            try
            {
                objApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            }
            catch (Exception ex)
            {
                // Get the type from the Solid Edge ProgID
                objType = Type.GetTypeFromProgID("SolidEdge.Application");
                // Start Solid Edge
                objApplication = Activator.CreateInstance(objType);
            }
            finally
            {
                // Make Solid Edge visible
                objApplication.Visible = true;
            }


            // open the document mentioned in the doc files.

            objDraftDoc = objApplication.ActiveDocument;
            if (objDraftDoc == null)
            {
                MessageBox.Show("Could not open document ");
            }

            Sheet = objDraftDoc.ActiveSheet;

            if (Sheet == null)
            {
                MessageBox.Show("No active sheet in the document");

            }

            objDimensions = Sheet.Dimensions;
            objDimension = objDimensions.Item(1);

            //****************DIMENSIONS*****************
            //***************unit type*******************
            objDimension.UnitsType = 1;

            int dimUnitType = objDimension.UnitsType;


            if (dimUnitType != 1)
            {
                MessageBox.Show("Failed to get/set unit type");

            }

            //***************Discrete values*******************
            double[] inputDiscreteValues = new double[4];
            double[] outputDiscreteValues = new double[1];

            inputDiscreteValues[0] = 0.0255;
            inputDiscreteValues[1] = 0.26;
            inputDiscreteValues[2] = 0.265;
            inputDiscreteValues[3] = 0.27;

            objDimension.SetValueDiscreteValues(inputDiscreteValues);

            objDimension.GetValueDiscreteValues(outputDiscreteValues);

            if (outputDiscreteValues[0] != inputDiscreteValues[0] || outputDiscreteValues[1] != inputDiscreteValues[1] || outputDiscreteValues[2] != inputDiscreteValues[2] || outputDiscreteValues[3] != inputDiscreteValues[3])
            {
                MessageBox.Show("Failed to get/set discrete values");

            }

            //***************Add discrete value*******************
            objDimension.AddDiscreteValue(0.22);

            objDimension.GetValueDiscreteValues(outputDiscreteValues);

            if (outputDiscreteValues[2] != 0.22)
            {
                MessageBox.Show("Failed to add discrete values");
            }

            //***************Remove discrete value*******************
            objDimension.RemoveDiscreteValue(0.265);

            objDimension.GetValueDiscreteValues(outputDiscreteValues);

            if (outputDiscreteValues[4] != 0.27)
            {
                MessageBox.Show("Failed to remove discrete values");
            }

            //***************Range value*******************
            objDimension.SetValueRangeValues(0.01, 5, 1);

            double LowValue = 0;
            double HighValue = 0;
            int condition = 0;

            objDimension.GetValueRangeValues(LowValue, condition, HighValue);

            if (LowValue != 0.01 || condition != 5 || HighValue != 1)
            {
                MessageBox.Show("Failed to set/get range values");
            }

            //***************Range value low*******************
            objDimension.SetValueRangeLowValue(0.001);

            objDimension.GetValueRangeValues(LowValue, condition, HighValue);

            if (LowValue != 0.001)
            {
                MessageBox.Show("Failed to set/get low range values");

            }

            //***************Range value high*******************
            objDimension.SetValueRangeHighValue(2);

            objDimension.GetValueRangeValues(LowValue, condition, HighValue);

            if (HighValue != 2)
            {
                MessageBox.Show("Failed to set/get high range values");

            }

            //*******************Get out of range value****************
            double dBadValue = 0;
            double dimensionValue = 3.0;

            //objDimension.Value = dimensionValue
            dBadValue = objDimension.GetValueOutOfRange();

            if (dBadValue != 0.02)
            {
                MessageBox.Show("Failed to get out of range value");

            }

            //*******************Clear limits or discrete values****************
            int count = 0;
            objDimension.ClearLimitsOrDiscreteValues();

            objDimension.GetValueDiscreteValues(outputDiscreteValues);
            outputDiscreteValues.GetLength(count);

            if (count != 0)
            {
                MessageBox.Show("Failed to clear limits or discrete values");
            }

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());

        }
        finally
        {

            // close the doc
            objDraftDoc.Close(false);
            objDraftDoc = null;
            objApplication = null;

        }


    }
}
