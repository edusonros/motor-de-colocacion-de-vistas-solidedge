using System;
using System.Runtime.InteropServices;

namespace Examples
{
    class Program
    {
        [STAThread]
        static void Main(string[] args)
        {
            SolidEdgeFramework.Application application = null;
            SolidEdgeFramework.Window window = null;
            SolidEdgeFramework.View view = null;

            try
            {
                // See "Handling 'Application is Busy' and 'Call was Rejected By Callee' errors" topic.
                OleMessageFilter.Register();

                // Attempt to connect to a running instance of Solid Edge.
                application = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");

                // A 3D document must be open.
                window = application.ActiveWindow as SolidEdgeFramework.Window;

                if (window != null)
                {
                    view = window.View;

                    double EyeX, EyeY, EyeZ, TarX, TarY, TarZ, UpX, UpY, UpZ, ScaleOrAngle;
                    bool Perspective;

                    view.GetCamera(out EyeX, out EyeY, out EyeZ, out TarX, out TarY, out TarZ, out UpX, out UpY, out UpZ, out Perspective, out ScaleOrAngle);
                }
            }
            catch (System.Exception ex)
            {
                Console.WriteLine(ex);
            }
            finally
            {
                OleMessageFilter.Unregister();
            }
        }
    }
}
