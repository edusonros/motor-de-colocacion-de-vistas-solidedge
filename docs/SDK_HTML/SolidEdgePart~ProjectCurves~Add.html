Imports System.Runtime.InteropServices

Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objRPlanes As SolidEdgePart.RefPlanes = Nothing
        Dim objPPlane(0 To 4) As SolidEdgePart.RefPlane
        Dim objRPParallel(0 To 16) As SolidEdgePart.RefPlane
        Dim objSketch As SolidEdgePart.Sketch = Nothing
        Dim objProfiles As SolidEdgePart.Profiles = Nothing
        Dim objProfile As SolidEdgePart.Profile = Nothing
        Dim objLine(0 To 4) As SolidEdgeFrameworkSupport.Line2d
        Dim objProjCurves As SolidEdgePart.ProjectCurves = Nothing
        Dim objProjCurve(0 To 4) As SolidEdgePart.ProjectCurve
        Dim Direction As SolidEdgeConstants.FeaturePropertyConstants
        Dim igProjOption As SolidEdgeConstants.FeaturePropertyConstants
        Dim profile_curve_body As SolidEdgeGeometry.CurveBody
        Dim objCurves As Object
        Dim profile_curve1 As Object = Nothing
        Dim profile_curve2 As Object = Nothing
        Dim profile_curve3 As Object = Nothing
        Const m = 39.37

        Try
            objApp = Marshal.GetActiveObject("SolidEdge.Application")

            ' Create/get the application with specific settings
            If objApp Is Nothing Then
                MsgBox("Could not get/create application")
                Exit Sub
            End If
            objDoc = objApp.ActiveDocument
            ' getting the reference planes collection object
            objRPlanes = objDoc.RefPlanes

            ' adding a reference planes parallel to and at a distance to a pricipal plane
            objPPlane(1) = objRPlanes.Item(1)

            objRPParallel(1) = objRPlanes.AddParallelByDistance(objPPlane(1), Distance:=0.6 / m, _
                NormalSide:=SolidEdgePart.ReferenceElementConstants.igReverseNormalSide)
            objRPParallel(2) = objRPlanes.AddParallelByDistance(objPPlane(1), Distance:=2 / m, _
                NormalSide:=SolidEdgePart.ReferenceElementConstants.igReverseNormalSide)
            objRPParallel(3) = objRPlanes.AddParallelByDistance(objPPlane(1), Distance:=3 / m, _
                NormalSide:=SolidEdgePart.ReferenceElementConstants.igReverseNormalSide)
            ' USER DISPLAY

            ' PASS/FAIL LOGIC
            If (objRPlanes.Count <> 6) Then
                MsgBox("AddParallelByDistance fails")
            End If

            ' Create a Sketch object
            objSketch = objDoc.Sketches.Add
            objProfiles = objSketch.Profiles
            objProfile = objProfiles.Add(objRPParallel(1))
            objLine(1) = objProfile.Lines2d.AddBy2Points(x1:=-0.6 / m, y1:=-0.6 / m, x2:=0.6 / m, y2:=0.6 / m)
            objLine(2) = objProfile.Lines2d.AddBy2Points(x1:=-0.6 / m, y1:=0.6 / m, x2:=0.6 / m, y2:=-0.6 / m)
            objLine(3) = objProfile.Lines2d.AddBy2Points(x1:=-0.8 / m, y1:=0.8 / m, x2:=-0.8 / m, y2:=-0.8 / m)

            objProjCurves = objDoc.Constructions.ProjectCurves

            ' Set the prjection along vector direction as specfied below
            igProjOption = SolidEdgeConstants.FeaturePropertyConstants.igProjectOptionProject

            ' Set the direction to project to i.e try projection in both directions
            Direction = SolidEdgeConstants.FeaturePropertyConstants.igBoth

            profile_curve_body = objProfile.CurveBody
            objCurves = profile_curve_body.Curves
            profile_curve1 = objCurves(1)
            profile_curve2 = objCurves(2)
            profile_curve3 = objCurves(3)

            ' Do a single curve
            objProjCurve(1) = objProjCurves.Add(profile_curve1, objPPlane(1), objPPlane(1), Direction, igProjOption)

            ' Do an entire sketch
            objProjCurve(2) = objProjCurves.Add(objSketch, objRPParallel(2), objRPParallel(2), Direction, igProjOption)

            ' Do an array of objects
            Dim arrCurves(1) As Object
            Dim arrPlanes(0) As Object
            arrCurves(0) = profile_curve2
            arrCurves(1) = profile_curve3
            arrPlanes(0) = objRPParallel(3)
            objProjCurve(3) = objProjCurves.AddArray(arrCurves, arrPlanes, objPPlane(1), Direction, igProjOption)

        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System.Windows.Forms;
using System.Runtime.InteropServices;

public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApp = null;
        SolidEdgePart.PartDocument objDoc = null;
        SolidEdgePart.RefPlanes objRPlanes = null;
        SolidEdgePart.RefPlane[] objPPlane = new SolidEdgePart.RefPlane[5];
        SolidEdgePart.RefPlane[] objRPParallel = new SolidEdgePart.RefPlane[17];
        SolidEdgePart.Sketch objSketch = null;
        SolidEdgePart.Profiles objProfiles = null;
        SolidEdgePart.Profile objProfile = null;
        SolidEdgeFrameworkSupport.Line2d[] objLine = new SolidEdgeFrameworkSupport.Line2d[5];
        SolidEdgePart.ProjectCurves objProjCurves = null;
        SolidEdgePart.ProjectCurve[] objProjCurve = new SolidEdgePart.ProjectCurve[5];
        SolidEdgeConstants.FeaturePropertyConstants Direction = null;
        SolidEdgeConstants.FeaturePropertyConstants igProjOption = null;
        SolidEdgeGeometry.CurveBody profile_curve_body = null;
        object objCurves = null;
        object profile_curve1 = null;
        object profile_curve2 = null;
        object profile_curve3 = null;
        const double m = 39.37;

        try
        {
            objApp = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");

            // Create/get the application with specific settings
            if (objApp == null)
            {
                MessageBox.Show("Could not get/create application");
                return;
            }
            objDoc = objApp.ActiveDocument;
            // getting the reference planes collection object
            objRPlanes = objDoc.RefPlanes;

            // adding a reference planes parallel to and at a distance to a pricipal plane
            objPPlane[1] = objRPlanes.Item(1);

            objRPParallel[1] = objRPlanes.AddParallelByDistance(objPPlane[1], Distance:0.6 / m, NormalSide:SolidEdgePart.ReferenceElementConstants.igReverseNormalSide);
            objRPParallel[2] = objRPlanes.AddParallelByDistance(objPPlane[1], Distance:2 / m, NormalSide:SolidEdgePart.ReferenceElementConstants.igReverseNormalSide);
            objRPParallel[3] = objRPlanes.AddParallelByDistance(objPPlane[1], Distance:3 / m, NormalSide:SolidEdgePart.ReferenceElementConstants.igReverseNormalSide);
            // USER DISPLAY

            // PASS/FAIL LOGIC
            if (objRPlanes.Count != 6)
            {
                MessageBox.Show("AddParallelByDistance fails");
            }

            // Create a Sketch object
            objSketch = objDoc.Sketches.Add;
            objProfiles = objSketch.Profiles;
            objProfile = objProfiles.Add(objRPParallel[1]);
            objLine[1] = objProfile.Lines2d.AddBy2Points(x1:-0.6 / m, y1:-0.6 / m, x2:0.6 / m, y2:0.6 / m);
            objLine[2] = objProfile.Lines2d.AddBy2Points(x1:-0.6 / m, y1:0.6 / m, x2:0.6 / m, y2:-0.6 / m);
            objLine[3] = objProfile.Lines2d.AddBy2Points(x1:-0.8 / m, y1:0.8 / m, x2:-0.8 / m, y2:-0.8 / m);

            objProjCurves = objDoc.Constructions.ProjectCurves;

            // Set the prjection along vector direction as specfied below
            igProjOption = SolidEdgeConstants.FeaturePropertyConstants.igProjectOptionProject;

            // Set the direction to project to i.e try projection in both directions
            Direction = SolidEdgeConstants.FeaturePropertyConstants.igBoth;

            profile_curve_body = objProfile.CurveBody;
            objCurves = profile_curve_body.Curves;
            profile_curve1 = objCurves(1);
            profile_curve2 = objCurves(2);
            profile_curve3 = objCurves(3);

            // Do a single curve
            objProjCurve[1] = objProjCurves.Add(profile_curve1, objPPlane[1], objPPlane[1], Direction, igProjOption);

            // Do an entire sketch
            objProjCurve[2] = objProjCurves.Add(objSketch, objRPParallel[2], objRPParallel[2], Direction, igProjOption);

            // Do an array of objects
            object[] arrCurves = new object[2];
            object[] arrPlanes = new object[1];
            arrCurves[0] = profile_curve2;
            arrCurves[1] = profile_curve3;
            arrPlanes[0] = objRPParallel[3];
            objProjCurve[3] = objProjCurves.AddArray(arrCurves, arrPlanes, objPPlane[1], Direction, igProjOption);

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }

    public Form1()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
