AnnotInitData as object
Description
Creates a SurfaceFinishSymbol. For creating PMI SurfaceFinishSymbol a plane must be set (XY plane is the default). The collection object from which it is derived (sheet/profile2d/PMI) dictate the creation of SurfaceFinishSymbol.

    Sub Main()
        Dim objApplication As SolidEdgeFramework.Application = Nothing
        Dim objPartDoc As SolidEdgePart.PartDocument = Nothing
        Dim objPMI As SolidEdgeFrameworkSupport.PMI = Nothing
        Dim objEdges As SolidEdgeGeometry.Edges = Nothing
        Dim objEdge1 As SolidEdgeGeometry.Edge = Nothing
        Dim objFolder As DirectoryInfo
        Dim objFileInfo As FileInfo
        Dim objGostWeldSymbol As SolidEdgeFrameworkSupport.GostWeldSymbol = Nothing
        Dim objGostWeldSymbols As SolidEdgeFrameworkSupport.GostWeldSymbols = Nothing
        Dim objSTS As SolidEdgeFrameworkSupport.SurfaceFinishSymbol = Nothing
        Dim objSTSs As SolidEdgeFrameworkSupport.SurfaceFinishSymbols = Nothing
        Dim objRefPlanes As SolidEdgePart.RefPlanes = Nothing
        Dim objAnnotPlane As SolidEdgePart.RefPlane = Nothing
        Dim objAnnotInitData As SolidEdgeFrameworkSupport.AnnotInitData = Nothing

        Try
            'Get the application with specific settings
            objApplication = Marshal.GetActiveObject("SolidEdge.Application")
            'Get Training folder path
            objFolder = GetTrainingFolder()
            objFileInfo = New FileInfo(Path.Combine(objFolder.FullName, "17_tooth_pinion.par"))
            'Open Part file 
            objPartDoc = objApplication.Documents.Open(objFileInfo.FullName)

            'Get the PMI object from the document
            objPartDoc.PMI_ByModelState(objPMI)

            'Create a reference (membassy) the edge
            objEdges = objPartDoc.Models.Item(1).ExtrudedProtrusions.Item(1).Edges(SolidEdgeConstants.FeatureTopologyQueryTypeConstants.igQueryAll)
            objEdge1 = objEdges.Item(1)

            objGostWeldSymbols = objPMI.GostWeldSymbols
            objSTSs = objPMI.SurfaceFinishSymbols

            objAnnotInitData = objGostWeldSymbols.AnnotInitData
            objAnnotInitData.ClearCreationData()

            objGostWeldSymbol = objGostWeldSymbols.Item(1)

            objAnnotPlane = objPartDoc.RefPlanes.Item(3)
            objAnnotInitData.SetPlane(objAnnotPlane)

            objAnnotInitData.SetTerminatorElement(objGostWeldSymbol, False, False, 0.1, 0.3, 0.2)
            objSTS = objSTSs.AddSurfaceFinishSymbol(AnnotInitData:=objAnnotInitData)

            objSTS.AddVertex(0.01, 0.02, 0.01)
            objSTS.Leader = True
            objSTS.BreakLine = True
            objSTS.RoughnessValue = "STS"
            objSTS.SurfaceFinishSymbol = SolidEdgeConstants.DimSurfTextureSymTypeConstants.igDimSurfaceFinishMachined

            
        Catch ex As Exception
            Console.WriteLine(ex.ToString)
        End Try

    End Sub
    Function GetTrainingFolder() As DirectoryInfo
        Dim objInstallData As SEInstallDataLib.SEInstallData = Nothing
        Dim objInstallFolder As DirectoryInfo = Nothing
        Dim objTrainingFolder As DirectoryInfo = Nothing

        Try
            objInstallData = New SEInstallDataLib.SEInstallData
            objInstallFolder = New DirectoryInfo(objInstallData.GetInstalledPath())
            objTrainingFolder = New DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"))
        Catch
        Finally
            If Not (objInstallData Is Nothing) Then
                Marshal.FinalReleaseComObject(objInstallData)
                objInstallData = Nothing
            End If
        End Try

        Return objTrainingFolder
    End Function
