using System.Windows.Forms;
using System.Runtime.InteropServices;
public class ExtendSurfaces
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgePart.PartDocument objPartDoc = null;
        SolidEdgeGeometry.Edges objEdges = null;
        object[] objFaces = new object[2];
        SolidEdgeGeometry.Edge objEdge = null;
        SolidEdgePart.ConstructionModel objConstructionModel = null;
        object[] objEdgArr = new object[4];
        dynamic objSurfaces = null;
        object[] objSurfaces1 = new object[2];
        SolidEdgeConstants.KeyPointType[] kptypeArr = new SolidEdgeConstants.KeyPointType[3];
        SolidEdgePart.Constructions objConstructions = null;
        double[] PositionArray = new double[9];
        SolidEdgeConstants.PointTypeConstants[] pointtypeArr = new SolidEdgeConstants.PointTypeConstants[3];
        double[] pointonface = new double[3];
        dynamic odjBodies = null;
        dynamic odjBodies1 = null;
        SolidEdgePart.ExtendSurface extendsurf = null;
        int nValidEdges = 0;
        object[] validEdges = new object[5];
        SolidEdgeGeometry.Body objTarget = null;
        SolidEdgeGeometry.Body objTarget1 = null;
        Type objType = null;

        try
        {
            // Create/get the application with specific settings
            objApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");

            if (objApplication == null)
            {
                // Get the type from the Solid Edge ProgID
                objType = Type.GetTypeFromProgID("SolidEdge.Application");
                // Start Solid Edge
                objApplication = Activator.CreateInstance(objType);
                // Make Solid Edge visible
                objApplication.Visible = true;
            }

            // Get the active document.
            objPartDoc = objApplication.ActiveDocument();
            if (Microsoft.VisualBasic.Information.Err().Number != 0 || objPartDoc == null)
            {
                MessageBox.Show("Could not open document ");

            }

            //obtain the first model and get all its edges

            objSurfaces = objPartDoc.Constructions.SurfaceByBoundaries.Item(1).Faces(SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryAll);
            odjBodies = objPartDoc.Constructions.ExtrudedSurfaces.Item(1).Faces(SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryAll);
            odjBodies1 = objPartDoc.Constructions.ExtrudedSurfaces.Item(2).Faces(SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryAll);

            objEdges = objSurfaces.Item(1).Edges;

            objTarget = odjBodies.Item(1).Body;
            objTarget1 = odjBodies1.Item(1).Body;

            //obtain constrction object
            objConstructions = objPartDoc.Constructions;
            if (Microsoft.VisualBasic.Information.Err().Number != 0 || objConstructions == null)
            {
                MessageBox.Show("Could not open document ");
            }

            objConstructionModel = objConstructions.Item(1);

            nValidEdges = 0;
            for (var nIndex = 1; nIndex <= objEdges.Count; nIndex++)
            {

                objEdge = objEdges(nIndex);

                validEdges[nValidEdges] = objEdge;
                nValidEdges = nValidEdges + 1;

                if (2 == nIndex)
                {
                    goto doit;
                }
            }

doit:
            extendsurf = objConstructionModel.ExtendSurfaces.Add1(nValidEdges, validEdges, 0, SolidEdgePart.ExtendSurfaceExtentTypeConstants.igESLinear, objTarget);
            extendsurf.Target = objTarget1;
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }

    public ExtendSurfaces()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
