    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApplication As SolidEdgeFramework.Application = Nothing
        Dim objPartDoc As SolidEdgePart.PartDocument = Nothing
        Dim objFaces As SolidEdgeGeometry.Faces = Nothing
        Dim objFace As SolidEdgeGeometry.Face = Nothing
        Dim objVertices As SolidEdgeGeometry.Vertices = Nothing
        Dim objVertex As SolidEdgeGeometry.Vertex = Nothing
        Dim objScaleBodyFeatures As SolidEdgePart.ScaleBodyFeatures = Nothing
        Dim objScaleBodyFeature As SolidEdgePart.ScaleBodyFeature = Nothing
        Dim objFeatArray(2) As Object
        Dim Name As String
        Dim x1 As Double
        Dim y1 As Double
        Dim z1 As Double
        Dim x2 As Double
        Dim y2 As Double
        Dim z2 As Double
        Dim xScale As Double
        Dim yScale As Double
        Dim zScale As Double
        Dim SEInstallDir As DirectoryInfo
        Try
            ' Get the application with specific settings
            objSEApplication = Marshal.GetActiveObject("SolidEdge.Application")
            'Get the path of Training folder
            SEInstallDir = GetTrainingFolder()
            'Open part file from Training Folder
            objPartDoc = objSEApplication.Documents.Open(SEInstallDir.FullName + "\17_tooth_pinion.par")
            'Get Scale Body Feature
            objScaleBodyFeatures = objPartDoc.Models.Item(1).ScaleBodyFeatures

            objFeatArray(0) = objPartDoc.Models.Item(1).Body
            'Get Faces
            objFaces = objPartDoc.Models.Item(1).Body.Faces(SolidEdgeConstants.FeatureTopologyQueryTypeConstants.igQueryAll)
            'Get the first face
            objFace = objFaces.Item(1)
            'Get the vertices from the first face
            objVertices = objFace.Vertices

            objVertex = objVertices.Item(1)
            'Add ScaleBodyFeature
            objScaleBodyFeature = objScaleBodyFeatures.Add(1, objFeatArray, 0.5, 0.5, 0.5, objVertex)

            If objScaleBodyFeature IsNot Nothing Then

                xScale = objScaleBodyFeature.UniformScale

                ' apply uniform scale of 2
                objScaleBodyFeature.UniformScale = 2

                ' set no key point
                objScaleBodyFeature.ScaleKeyPoint() = Nothing

                ' apply non-uniform scale
                objScaleBodyFeature.put_NonUniformScale(1, 2, 3)

                ' get non-uniform scale        
                objScaleBodyFeature.get_NonUniformScale(xScale, yScale, zScale)

                ' suppress scale feature   
                objScaleBodyFeature.Suppress = True
             
                objScaleBodyFeature.Suppress = False

                objScaleBodyFeature.Range(x1, y1, z1, x2, y2, z2)
              
                Name = objScaleBodyFeature.Name()

                ' delete scale feature
                objScaleBodyFeature.Delete()
            Else
                MsgBox("Could not get objScaleBodyFeature object")
            End If

         
        Catch ex As Exception
            MsgBox(ex.ToString)
        Finally
            objPartDoc.Close(False)
            objSEApplication = Nothing
            objPartDoc = Nothing
        End Try
    End Sub
    Function GetTrainingFolder() As DirectoryInfo
        Dim objInstallData As SEInstallDataLib.SEInstallData = Nothing
        Dim objInstallFolder As DirectoryInfo = Nothing
        Dim objTrainingFolder As DirectoryInfo = Nothing

        Try
            objInstallData = New SEInstallDataLib.SEInstallData
            objInstallFolder = New DirectoryInfo(objInstallData.GetInstalledPath())
            objTrainingFolder = New DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"))
        Catch
        Finally
            If Not (objInstallData Is Nothing) Then
                Marshal.FinalReleaseComObject(objInstallData)
                objInstallData = Nothing
            End If
        End Try

        Return objTrainingFolder
    End Function

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
using System.IO;
public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objSEApplication = null;
        SolidEdgePart.PartDocument objPartDoc = null;
        SolidEdgeGeometry.Faces objFaces = null;
        SolidEdgeGeometry.Face objFace = null;
        SolidEdgeGeometry.Vertices objVertices = null;
        SolidEdgeGeometry.Vertex objVertex = null;
        SolidEdgePart.ScaleBodyFeatures objScaleBodyFeatures = null;
        SolidEdgePart.ScaleBodyFeature objScaleBodyFeature = null;
        object[] objFeatArray = new object[3];
        string Name = null;
        double x1 = 0;
        double y1 = 0;
        double z1 = 0;
        double x2 = 0;
        double y2 = 0;
        double z2 = 0;
        double xScale = 0;
        double yScale = 0;
        double zScale = 0;
        DirectoryInfo SEInstallDir = null;
        try
        {
            // Get the application with specific settings
            objSEApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            //Get the path of Training folder
            SEInstallDir = GetTrainingFolder();
            //Open part file from Training Folder
            objPartDoc = objSEApplication.Documents.Open(SEInstallDir.FullName + "\\17_tooth_pinion.par");
            //Get Scale Body Feature
            objScaleBodyFeatures = objPartDoc.Models.Item(1).ScaleBodyFeatures;

            objFeatArray[0] = objPartDoc.Models.Item(1).Body;
            //Get Faces
            objFaces = objPartDoc.Models.Item(1).Body.Faces(SolidEdgeConstants.FeatureTopologyQueryTypeConstants.igQueryAll);
            //Get the first face
            objFace = objFaces.Item(1);
            //Get the vertices from the first face
            objVertices = objFace.Vertices;

            objVertex = objVertices.Item(1);
            //Add ScaleBodyFeature
            objScaleBodyFeature = objScaleBodyFeatures.Add(1, objFeatArray, 0.5, 0.5, 0.5, objVertex);

            if (objScaleBodyFeature != null)
            {

                xScale = objScaleBodyFeature.UniformScale;

                // apply uniform scale of 2
                objScaleBodyFeature.UniformScale = 2;

                // set no key point
                objScaleBodyFeature.ScaleKeyPoint() = null;

                // apply non-uniform scale
                objScaleBodyFeature.put_NonUniformScale(1, 2, 3);

                // get non-uniform scale        
                objScaleBodyFeature.get_NonUniformScale(xScale, yScale, zScale);

                // suppress scale feature   
                objScaleBodyFeature.Suppress = true;

                objScaleBodyFeature.Suppress = false;

                objScaleBodyFeature.Range(x1, y1, z1, x2, y2, z2);

                Name = objScaleBodyFeature.Name;

                // delete scale feature
                objScaleBodyFeature.Delete();
            }
            else
            {
                MessageBox.Show("Could not get objScaleBodyFeature object");
            }


        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
        finally
        {
            objPartDoc.Close(false);
            objSEApplication = null;
            objPartDoc = null;
        }
    }
    public DirectoryInfo GetTrainingFolder()
    {
        SEInstallDataLib.SEInstallData objInstallData = null;
        DirectoryInfo objInstallFolder = null;
        DirectoryInfo objTrainingFolder = null;

        try
        {
            objInstallData = new SEInstallDataLib.SEInstallData();
            objInstallFolder = new DirectoryInfo(objInstallData.GetInstalledPath());
            objTrainingFolder = new DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"));
        }
        catch
        {
        }
        finally
        {
            if (objInstallData != null)
            {
                Marshal.FinalReleaseComObject(objInstallData);
                objInstallData = null;
            }
        }

        return objTrainingFolder;
    }



    public Form1()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
