Specifies the values used to create the coordinate system.  This is a single-dimension Double array that contains 16 elements.
Description
Adds a coordinate system based on a 4x4 matrix of values.
Syntax
Visual Basic
Public Function AddByMatrix( _
   ByRef Matrix() As Double _
) As CoordinateSystem
Parameters
Matrix
Specifies the values used to create the coordinate system.  This is a single-dimension Double array that contains 16 elements.
Remarks

XPerspective, YPerspective, and ZPerspective must all be set to 0. Scale must be set to 1.

The matrix consists of

    (             , Px )
    (      R      , Py )
    (             , Pz )
    ( Tx, Ty, Tz,   S  )

R = a non singular transformation matrix. This matrix contains the rotation, reflection, non-uniform scaling and shearing components. (In our case it is used for rotation)

X-Rotation Transform
 CSTransform(0) = 2
 CSTransform(1) = 0
 CSTransform(2) = 0

Y-Rotation Transform
 CSTransform(4) = 0
 CSTransform(5) = 2
 CSTransform(6) = 0

Z-Rotation Transform
 CSTransform(8) = 0
 CSTransform(9) = 0
 CSTransform(10) = 2

T = a translation vector.
Tx -  CSTransform(12) - xTranslation
Ty -  CSTransform(13) - yTranslation
Tz -  CSTransform(14) - zTranslation

P = perspective terms in viewing transformations. Must be zero in transformations used for modelling.

Px - CSTransform(3) - Should be zero for modeling
Py - CSTransform(7) - Should be zero for modeling
Pz - CSTransform(11) - Should be zero for modeling

S = a global scaling factor. It has to be greater than zero. Its value is the inverse of the global scale.
 S - CSTransform(15) = 1 (ideally should be 1).

Imports System.Runtime.InteropServices
Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApplication As SolidEdgeFramework.Application = Nothing
        Dim objPartDoc As SolidEdgePart.PartDocument = Nothing
        Dim CS As SolidEdgePart.CoordinateSystem = Nothing

        Try
            objSEApplication = Marshal.GetActiveObject("SolidEdge.Application")
            objPartDoc = objSEApplication.Documents.Add("SolidEdge.PartDocument")
            objPartDoc.CoordinateSystems.Item(1).Visible = False
            Dim CSTransform(15) As Double
            CSTransform(0) = 2
            CSTransform(1) = 0  'X-Rotation Transform
            CSTransform(2) = 0

            CSTransform(3) = 0 'Px

            CSTransform(4) = 0
            CSTransform(5) = 2 'Y-Rotation Transform
            CSTransform(6) = 0

            CSTransform(7) = 0 'Py

            CSTransform(8) = 0
            CSTransform(9) = 0  'Z-Rotation Transform
            CSTransform(10) = 2

            CSTransform(11) = 0 'Pz

            CSTransform(12) = 0.05 Or 0.1 'xTranslation
            CSTransform(13) = 0.05 Or 0.1 'yTranslation
            CSTransform(14) = 0.05 Or 0.1 'zTranslation

            CSTransform(15) = 1

            CS = objPartDoc.CoordinateSystems.AddByMatrix(CSTransform)


        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objSEApplication = null;
        SolidEdgePart.PartDocument objPartDoc = null;
        SolidEdgePart.CoordinateSystem CS = null;

        try
        {
            objSEApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            objPartDoc = objSEApplication.Documents.Add("SolidEdge.PartDocument");
            objPartDoc.CoordinateSystems.Item(1).Visible = false;
            double[] CSTransform = new double[16];
            CSTransform[0] = 2;
            CSTransform[1] = 0; //X-Rotation Transform
            CSTransform[2] = 0;

            CSTransform[3] = 0; //Px

            CSTransform[4] = 0;
            CSTransform[5] = 2; //Y-Rotation Transform
            CSTransform[6] = 0;

            CSTransform[7] = 0; //Py

            CSTransform[8] = 0;
            CSTransform[9] = 0; //Z-Rotation Transform
            CSTransform[10] = 2;

            CSTransform[11] = 0; //Pz

            CSTransform[12] = 0.05 | 0.1; //xTranslation
            CSTransform[13] = 0.05 | 0.1; //yTranslation
            CSTransform[14] = 0.05 | 0.1; //zTranslation

            CSTransform[15] = 1;

            CS = objPartDoc.CoordinateSystems.AddByMatrix(CSTransform);


        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }

    public Form1()
    {
        SubscribeToEvents();
    }

//NOTE: Converted event handler wireups:
    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
        Button1.Click += Button1_Click;
    }

}
