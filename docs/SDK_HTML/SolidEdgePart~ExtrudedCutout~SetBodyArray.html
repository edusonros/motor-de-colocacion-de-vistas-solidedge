            Else
                objProfile4(0).Visible = False
            End If

            nBodies = objDoc.Models.Count()

            ParentBodies = New SolidEdgeGeometry.Body(nBodies) {}
            NewBodies = New SolidEdgeGeometry.Body(nBodies) {}

            For ii = 0 To (nBodies - 1)
                objBody = objDoc.Models.Item(ii + 1).Body

                NewBodies(ii) = objBody
            Next ii

            IsMBC = True

            Call objExtCut.SetBodyArray(MultiBodyCut:=IsMBC, NumberOfBodies:=nBodies, BodyArray:=NewBodies)



            GetBodies = New SolidEdgeGeometry.Body(nBodies) {}

            objExtCut.GetBodyArray(MultiBodyCut:=IsMBC, NumberOfBodies:=nParentBodies, BodyArray:=ParentBodies)

            If IsMBC <> True Then
                MsgBox(("SetBodyArray Method failed to set MultiBodyCut flag TRUE"))
            End If

            If nParentBodies <> nBodies Then
                MsgBox(("SetBodyArray Method failed with wrong number of bodies"))
            End If

            For ii = 0 To nParentBodies - 1
                If ParentBodies(ii) Is Nothing Then
                    MsgBox(("SetBodyArray Method failed with NULL body"))
                End If
            Next ii

        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try

    End Sub
