Remarks
The properties returned are the current properties stored with the model, which means they can be not computed yet, out-of-date, or up-to-date. To calculate properties for the first time or to update the properties use the ComputePhysicalProperties method. Properties must not be user-overridden. To determine whether physical properties status, use the PhysicalPropertiesStatus property.

using System;
using System.Collections.Generic;
using System.Runtime.InteropServices;

namespace Examples
{
    class Program
    {
        [STAThread]
        static void Main(string[] args)
        {
            SolidEdgeFramework.Application application = null;
            SolidEdgePart.PartDocument partDocument = null;
            SolidEdgePart.Models models = null;
            SolidEdgePart.Model model = null;
            double density = 0;
            double accuracy = 0;
            double volume = 0;
            double area = 0;
            double mass = 0;
            Array cetnerOfGravity = Array.CreateInstance(typeof(double), 3);
            Array centerOfVolumne = Array.CreateInstance(typeof(double), 3);
            Array globalMomentsOfInteria = Array.CreateInstance(typeof(double), 6);     // Ixx, Iyy, Izz, Ixy, Ixz and Iyz 
            Array principalMomentsOfInteria = Array.CreateInstance(typeof(double), 3);  // Ixx, Iyy and Izz
            Array principalAxes = Array.CreateInstance(typeof(double), 9);              // 3 axes x 3 coords
            Array radiiOfGyration = Array.CreateInstance(typeof(double), 9);            // 3 axes x 3 coords
            double relativeAccuracyAchieved = 0;
            int status = 0;

            try
            {
                // See "Handling 'Application is Busy' and 'Call was Rejected By Callee' errors" topic.
                OleMessageFilter.Register();

                // Attempt to connect to a running instance of Solid Edge.
                application = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
                partDocument = application.ActiveDocument as SolidEdgePart.PartDocument;

                if (partDocument != null)
                {
                    models = partDocument.Models;
                    model = models.Item(1);

                    model.GetPhysicalProperties(
                         Status: out status,
                         Density: out density,
                         Accuracy: out accuracy,
                         Volume: out volume,
                         Area: out area,
                         Mass: out mass,
                         CenterOfGravity: ref cetnerOfGravity,
                         CenterOfVolume: ref centerOfVolumne,
                         GlobalMomentsOfInteria: ref globalMomentsOfInteria,
                         PrincipalMomentsOfInteria: ref principalMomentsOfInteria,
                         PrincipalAxes: ref principalAxes,
                         RadiiOfGyration: ref radiiOfGyration,
                         RelativeAccuracyAchieved: out relativeAccuracyAchieved);
                }
            }
            catch (System.Exception ex)
            {
                Console.WriteLine(ex);
            }
            finally
            {
                OleMessageFilter.Unregister();
            }
        }
    }
}

static double lGetMass(ModelsPtr pModels, PartDocumentPtr pPartDoc,  bool bUserDefineProps)
{
    HRESULT hr = S_OK;
    int nStatus = 0;
    double dDensity = 0;
    double dAccuracy = 0;
    double dVolume = 0;
    double dArea = 0;
    double dMass = 0;
    SAFEARRAY* saCG = NULL;
    SAFEARRAY* saCV = NULL;
    SAFEARRAY* saMI = NULL;
    SAFEARRAY* saPI = NULL;
    SAFEARRAY* saPA = NULL;
    SAFEARRAY* saRG = NULL;
    double dAccuracyAchived = 0;
    ModelPtr pModel = 0;

    if (bUserDefineProps)
    {
        hr = pPartDoc->GetUserPhysicalProperties(
                &dVolume, &dArea, &dMass,
                &saCG,
                &saCV,
                &saMI,
                &saPI,
                &saPA,
                &saRG
            );
    }
    else
    {
        if (!pModels || (pModels->Count == 0))
            goto wrapup;

        pModel = pModels->Item(1);

        hr = pModel->GetPhysicalProperties(&nStatus, &dDensity, &dAccuracy,
            &dVolume, &dArea, &dMass,
            &saCG,
            &saCV,
            &saMI,
            &saPI,
            &saPA,
            &saRG, 
            &dAccuracyAchived);
    }

wrapup:

    return dMass;
}

void PartSMMassProperties(SolidEdgeDocumentPtr pDoc)
{    
    SolidEdgeFramework::DocumentTypeConstants docType = SolidEdgeFramework::DocumentTypeConstants::igUnknownDocument;
    PartDocumentPtr pPartDoc = 0;
    
    if(pDoc)
    { 
        docType = pDoc->Type;
    }

    // we are only looking for Part document type but it is the same for SM document
    if (SolidEdgeFramework::DocumentTypeConstants::igPartDocument == docType)
    {
        pPartDoc = pDoc;
        SolidEdgePart::PhysicalPropertiesStatusConstants phyPropStatus = pPartDoc->PhysicalPropertiesStatus;

        switch (phyPropStatus)
        {
            case SolidEdgePart::PhysicalPropertiesStatusConstants::sePhysicalPropertiesStatus_None:
            {
                ::MessageBox(NULL, L"Document does not have mass properties computed.\n Please set a Material and then go into Inspect->Physical Properties dialog to compute the mass.", L"Get Mass From Part Document Sample", MB_OK);
                break;
            }
            case SolidEdgePart::PhysicalPropertiesStatusConstants::sePhysicalPropertiesStatus_Model:
            {    
                ModelsPtr pModels = pPartDoc->Models;
                lGetMass(pModels, pPartDoc, false);
                break;
            }
            case SolidEdgePart::PhysicalPropertiesStatusConstants::sePhysicalPropertiesStatus_User:
            {
                lGetMass(0, pPartDoc, true);
                break;
            }
        }
    }
    else
    {
        goto wrapup;
    }
    
wrapup:

    return;
}
