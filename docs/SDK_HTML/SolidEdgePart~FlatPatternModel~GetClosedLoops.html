    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objApplication As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.SheetMetalDocument = Nothing
        Dim objFPM As SolidEdgePart.FlatPatternModel = Nothing
        Dim objHS As SolidEdgeFramework.HighlightSet = Nothing
        Dim AEdges As SolidEdgeGeometry.Edges = Nothing
        Dim objEdge As SolidEdgeGeometry.Edge = Nothing
        Dim ii As Long
        Dim jj As Long
        Dim NumberOfLayer1Loops As Long
        Dim NumberOfLayer2Loops As Long
        Dim Layer1LoopArray(0 To 1) As Object
        Dim Layer2LoopArray(0 To 1) As Object
        Dim RefKey() As Byte

        Try

            objApplication = GetObject(, "SolidEdge.Application")
            objDoc = objApplication.ActiveDocument

            objHS = objDoc.HighlightSets.Add

            objHS.Color = System.Drawing.Color.Magenta.ToKnownColor()

            ReDim RefKey(32)

            objFPM = objDoc.FlatPatternModels.Item(1)

            If (objFPM IsNot Nothing) Then

                Call objFPM.GetClosedLoops(NumberOfLayer1Loops, Layer1LoopArray, NumberOfLayer2Loops, Layer2LoopArray)

                If (NumberOfLayer1Loops = NumberOfLayer2Loops) Then
                    For ii = 0 To (NumberOfLayer1Loops - 1)
                        AEdges = Layer1LoopArray(ii)

                        For jj = 1 To AEdges.Count
                            Call AEdges.Item(jj).GetReferenceKey(RefKey)

                            Call objDoc.BindKeyToObject(RefKey, objEdge)

                            Call objHS.AddItem(objEdge)

                            objHS.Draw()
                        Next

                        AEdges = Layer2LoopArray(ii)

                        For jj = 1 To AEdges.Count
                            Call AEdges.Item(jj).GetReferenceKey(RefKey)

                            Call objDoc.BindKeyToObject(RefKey, objEdge)

                            Call objHS.AddItem(objEdge)

                            objHS.Draw()
                        Next

                        jj = 0
                    Next
                Else
                    For ii = 0 To (NumberOfLayer1Loops - 1)
                        AEdges = Layer1LoopArray(ii)

                        For jj = 1 To AEdges.Count
                            Call AEdges.Item(jj).GetReferenceKey(RefKey)

                            Call objDoc.BindKeyToObject(RefKey, objEdge)

                            Call objHS.AddItem(objEdge)

                            objHS.Draw()
                        Next
                    Next

                    For ii = 0 To (NumberOfLayer2Loops - 1)
                        AEdges = Layer2LoopArray(ii)

                        For jj = 1 To AEdges.Count
                            Call AEdges.Item(jj).GetReferenceKey(RefKey)

                            Call objDoc.BindKeyToObject(RefKey, objEdge)

                            Call objHS.AddItem(objEdge)

                            objHS.Draw()
                        Next
                    Next
                End If

                MsgBox("Done")
            Else
                MsgBox("No Flat Pattern")
            End If

        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
using System.IO;

public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgePart.SheetMetalDocument objDoc = null;
        SolidEdgePart.FlatPatternModel objFPM = null;
        SolidEdgeFramework.HighlightSet objHS = null;
        SolidEdgeGeometry.Edges AEdges = null;
        SolidEdgeGeometry.Edge objEdge = null;
        long ii = 0;
        long jj = 0;
        long NumberOfLayer1Loops = 0;
        long NumberOfLayer2Loops = 0;
        object[] Layer1LoopArray = new object[2];
        object[] Layer2LoopArray = new object[2];
        byte[] RefKey = null;

        try
        {

            objApplication = Microsoft.VisualBasic.Interaction.GetObject(null, "SolidEdge.Application");
            objDoc = objApplication.ActiveDocument;

            objHS = objDoc.HighlightSets.Add;

            objHS.Color = System.Drawing.Color.Magenta.ToKnownColor();

            RefKey = new byte[33];

            objFPM = objDoc.FlatPatternModels.Item(1);

            if (objFPM != null)
            {

                objFPM.GetClosedLoops(NumberOfLayer1Loops, Layer1LoopArray, NumberOfLayer2Loops, Layer2LoopArray);

                if (NumberOfLayer1Loops == NumberOfLayer2Loops)
                {

                    long tempVar = NumberOfLayer1Loops - 1;
                    for (ii = 0; ii <= tempVar; ii++)
                    {
                        AEdges = (SolidEdgeGeometry.Edges)Layer1LoopArray[ii];

                        for (jj = 1; jj <= AEdges.Count; jj++)
                        {
                            AEdges.Item(jj).GetReferenceKey(RefKey);

                            objDoc.BindKeyToObject(RefKey, objEdge);

                            objHS.AddItem(objEdge);

                            objHS.Draw();
                        }

                        AEdges = (SolidEdgeGeometry.Edges)Layer2LoopArray[ii];

                        for (jj = 1; jj <= AEdges.Count; jj++)
                        {
                            AEdges.Item(jj).GetReferenceKey(RefKey);

                            objDoc.BindKeyToObject(RefKey, objEdge);

                            objHS.AddItem(objEdge);

                            objHS.Draw();
                        }

                        jj = 0;
                    }
                }
                else
                {

                    long tempVar2 = NumberOfLayer1Loops - 1;
                    for (ii = 0; ii <= tempVar2; ii++)
                    {
                        AEdges = (SolidEdgeGeometry.Edges)Layer1LoopArray[ii];

                        for (jj = 1; jj <= AEdges.Count; jj++)
                        {
                            AEdges.Item(jj).GetReferenceKey(RefKey);

                            objDoc.BindKeyToObject(RefKey, objEdge);

                            objHS.AddItem(objEdge);

                            objHS.Draw();
                        }
                    }


                    long tempVar3 = NumberOfLayer2Loops - 1;
                    for (ii = 0; ii <= tempVar3; ii++)
                    {
                        AEdges = (SolidEdgeGeometry.Edges)Layer2LoopArray[ii];

                        for (jj = 1; jj <= AEdges.Count; jj++)
                        {
                            AEdges.Item(jj).GetReferenceKey(RefKey);

                            objDoc.BindKeyToObject(RefKey, objEdge);

                            objHS.AddItem(objEdge);

                            objHS.Draw();
                        }
                    }
                }

                MessageBox.Show("Done");
            }
            else
            {
                MessageBox.Show("No Flat Pattern");
            }

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }
}
