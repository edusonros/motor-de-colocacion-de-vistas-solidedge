Imports System.Runtime.InteropServices
Imports System.IO

Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click

        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objPPart As SolidEdgePart.PatternPart = Nothing
        Dim objEdges(0 To 4) As Object
        Dim objEdges2(0 To 4) As Object
        Dim objRetEdges(0 To 10) As Object
        Dim objVertex As Object = Nothing
        Dim objVertex2 As Object
        Dim objFeatures(0 To 4) As Object
        Dim objPlane As SolidEdgePart.RefPlane = Nothing
        Dim anchorside As SolidEdgeConstants.PatternCurveAnchorSideConstants = Nothing
        Dim patterntype As SolidEdgeConstants.PatternOffsetTypeConstants = Nothing
        Dim transform As SolidEdgeConstants.PatternTransformTypeConstants = Nothing
        Dim rotatetype As SolidEdgeConstants.PatternTransformRotateTypeConstants = Nothing


        Try
            ' Create/get the application with specific settings
            objApp = Marshal.GetActiveObject("SolidEdge.Application")

            ' Create a part document with simple box to use this sample.
            objDoc = objApp.ActiveDocument

            Dim objExtProt As SolidEdgePart.ExtrudedProtrusion = objDoc.Models.Item(1).ExtrudedProtrusions.Item(1)


            Dim objEdgs As SolidEdgeGeometry.Edges = objExtProt.Edges(SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryAll)
            objEdges(0) = objEdgs.Item(1)

            objVertex = objEdges(0).StartVertex
            objVertex2 = objEdges(0).StartVertex

            objFeatures(0) = objDoc.Models.Item(1).ExtrudedProtrusions.Item(1)
            objPlane = objDoc.RefPlanes.Item(1)


            objPPart = objDoc.Models.Item(1).PatternParts.AddByCurve(1, objEdges, objVertex, SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveRightSide, 0, SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternFixedOffset, 5, 0.2, _
                   0, objEdges2, objVertex2, SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveRightSide, 0, SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternFitOffset, 0, 0.5, _
                   SolidEdgePart.PatternTransformTypeConstants.sePatternTransformLinear, SolidEdgePart.PatternTransformRotateTypeConstants.sePatternTransformRotateOnCurvePosition, objPlane)


        Catch ex As Exception
            MsgBox(ex.ToString)
        Finally

        End Try

    End Sub

End Class
