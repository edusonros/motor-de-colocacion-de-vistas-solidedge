using System.Windows.Forms;
using System.Runtime.InteropServices;

public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgeFramework.SolidEdgeInsightXT objSolidEdgeInsightXT = null;
        string FileItemID = null;
        string FileItemRevID = null;
        string FileDataSetName = null;
        object listoffiles = null;

        try
        {
            objApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            objSolidEdgeInsightXT = objApplication.SolidEdgeInsightXT;
            FileItemID = "PRT-03311";
            FileItemRevID = "A";
            FileDataSetName = "PRT-03311_A.par";

            objSolidEdgeInsightXT.GetMappedPropertiesForGivenFile(FileItemID, FileItemRevID, FileDataSetName, listoffiles);
            MessageBox.Show("Material of the part " + FileDataSetName + " is " + ((object[,])listoffiles)[1, 1].ToString());
            objSolidEdgeInsightXT = null;
            objApplication = null;
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
        finally
        {
            objApplication = null;
            objSolidEdgeInsightXT = null;
        }
    }

    public Form1()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
