Imports System.Runtime.InteropServices
Imports System.IO

Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objAppRet As SolidEdgeFramework.Application
        Dim objDoc As SolidEdgePart.PartDocument
        Dim objConstructions As SolidEdgePart.Constructions
        Dim objConstructionModel As SolidEdgePart.CopyConstructions
        Dim objModel As SolidEdgePart.CopyConstruction

        Dim bDeleteVoids As Boolean
        Dim ConfigurationName As String

        objDoc = Nothing
        Dim strDocFiles As String = "C:\Part1.par"

        Try

            ' Create/get the application with specific settings
            objAppRet = Marshal.GetActiveObject("SolidEdge.Application")

            objDoc = objAppRet.Documents.Add("SolidEdge.PartDocument")

            ' Set Ordered Mode
            objDoc.ModelingMode = SolidEdgePart.ModelingModeConstants.seModelingModeOrdered

            ' get the model collection
            objConstructions = objDoc.Constructions

            objConstructionModel = objConstructions.CopyConstructions

            objConstructionModel.AddEx(strDocFiles, 2, 2, 2, Nothing, Nothing, Nothing, Nothing, Nothing, Nothing, Nothing, Nothing, Nothing, True, "newconfig", True)

            objModel = objConstructions.CopyConstructions.Item(1)

            ' need to call update to reflect on display.
            objModel.Update()

            ' Get configuration name 
            ConfigurationName = objModel.ConfigName

            objModel.ConfigName = "newconfig2"

            objModel.Update()

            ' set configuration name
            ConfigurationName = objModel.ConfigName

        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
using System.IO;

public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objAppRet = null;
        SolidEdgePart.PartDocument objDoc = null;
        SolidEdgePart.Constructions objConstructions = null;
        SolidEdgePart.CopyConstructions objConstructionModel = null;
        SolidEdgePart.CopyConstruction objModel = null;

        bool bDeleteVoids = false;
        string ConfigurationName = null;

        objDoc = null;
        string strDocFiles = "C:\\Part1.par";

        try
        {

            // Create/get the application with specific settings
            objAppRet = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");

            objDoc = objAppRet.Documents.Add("SolidEdge.PartDocument");

            // Set Ordered Mode
            objDoc.ModelingMode = SolidEdgePart.ModelingModeConstants.seModelingModeOrdered;

            // get the model collection
            objConstructions = objDoc.Constructions;

            objConstructionModel = objConstructions.CopyConstructions;

            objConstructionModel.AddEx(strDocFiles, 2, 2, 2, null, null, null, null, null, null, null, null, null, true, "newconfig", true);

            objModel = objConstructions.CopyConstructions.Item(1);

            // need to call update to reflect on display.
            objModel.Update();

            // Get configuration name 
            ConfigurationName = objModel.ConfigName;

            objModel.ConfigName = "newconfig2";

            objModel.Update();

            // set configuration name
            ConfigurationName = objModel.ConfigName;

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }

    public Form1()
    {
        SubscribeToEvents();
    }

//Converted event handler wireups:
    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
        Button1.Click += Button1_Click;
        Button1.Click += Button1_Click;
        Button1.Click += Button1_Click;
        Button1.Click += Button1_Click;
    }

}
