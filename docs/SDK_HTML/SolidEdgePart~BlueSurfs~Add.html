using System.Windows.Forms;
using System.Runtime.InteropServices;
public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApp = null;
        SolidEdgePart.PartDocument objDoc = null;
        SolidEdgePart.Profile[] objProfile = new SolidEdgePart.Profile[3];
        SolidEdgeFrameworkSupport.Lines2d objLines = null;
        SolidEdgePart.Constructions objConstructions = null;
        SolidEdgePart.BlueSurfs objBlueSurfs = null;
        SolidEdgePart.BlueSurf objBlueSurf = null;
        object[] objCSections = new object[2];
        object[] objOrigins = new object[2];
        object[] objGuideCurves = new object[2];

        try
        {
            // Create/get the application with specific settings
            objApp = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            if (objApp == null)
            {
                objApp = System.Activator.CreateInstance(System.Type.GetTypeFromProgID("SolidEdge.Application"));
                objApp.Visible = true;
            }
            objDoc = objApp.Documents.Add("SolidEdge.PartDocument");
            //Draw the Base Profile
            objProfile[1] = objDoc.Sketches.Add.Profiles.Add(objDoc.RefPlanes.Item(3));
            objLines = objProfile[1].Lines2d;
            objLines.AddBy2Points(x1:0, y1:0, x2:0.08, y2:0);

            objProfile[2] = objDoc.Sketches.Add.Profiles.Add(objDoc.RefPlanes.Item(3));
            objLines = objProfile[2].Lines2d;
            objLines.AddBy2Points(x1:0, y1:0.06, x2:0.08, y2:0.06);
            objConstructions = objDoc.Constructions;
            objBlueSurfs = objConstructions.BlueSurfs;

            objCSections[0] = objDoc.Sketches.Item(1).Profile.CurveBody.Curves;
            objCSections[1] = objDoc.Sketches.Item(2).Profile.CurveBody.Curves;
            //
            //
            objOrigins[0] = objDoc.Sketches.Item(1).Profile.CurveBody.CurveVertices.Item(1);
            objOrigins[1] = objDoc.Sketches.Item(2).Profile.CurveBody.CurveVertices.Item(1);

            //
            objGuideCurves[1] = null;
            objGuideCurves[1] = null;
            //
            //
            objBlueSurf = objBlueSurfs.Add(2, objCSections, objOrigins, SolidEdgePart.FeaturePropertyConstants.igNatural, 0.0D, SolidEdgePart.FeaturePropertyConstants.igNatural, 0.0D, 0, objGuideCurves, SolidEdgePart.FeaturePropertyConstants.igNatural, 0.0D, SolidEdgePart.FeaturePropertyConstants.igNatural, 0.0D, false, false);
            //
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }

        // Release objects
        objApp = null;
        objDoc = null;


    }

    public Form1()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
