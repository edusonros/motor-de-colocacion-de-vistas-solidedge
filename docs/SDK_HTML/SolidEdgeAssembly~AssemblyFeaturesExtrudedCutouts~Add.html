Imports SolidEdgeFramework
Imports SolidEdgeAssembly
Imports SolidEdgePart
Imports SolidEdgeGeometry
Imports System.IO
Imports System.Runtime.InteropServices


Public Class Form1
    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApplication As SolidEdgeFramework.Application = Nothing

        Dim objDoc As AssemblyDocument

        Dim objPartDoc As PartDocument = Nothing
        Dim objOccs As Occurrences = Nothing
        Dim objOcc As Occurrence = Nothing
        Dim objModel As Model = Nothing
        Dim objSketchs As Sketchs = Nothing
        Dim objSketch As Sketch = Nothing

        Dim objExtrCOs As ExtrudedCutouts = Nothing
        Dim objExtrCO As ExtrudedCutout = Nothing
        Dim objExtrCOProf As Profile = Nothing
        Dim objExtrPRs As ExtrudedProtrusions = Nothing
        Dim objExtrPR As ExtrudedProtrusion = Nothing
        Dim objExtrPRProf As Profile = Nothing
        Dim objFaceRef As Reference = Nothing
        Dim objEdgeRef As Reference = Nothing
        Dim objEdges As Edges = Nothing
        Dim objEdge As Edge = Nothing
        Dim objFaces As Faces = Nothing
        Dim objFace As Face = Nothing

        Dim objAsmRefPlanes As AsmRefPlanes = Nothing
        Dim objAsmRefPlane As AsmRefPlane = Nothing
        Dim objLayouts As Layouts = Nothing
        Dim objLayout As Layout = Nothing
        Dim objLayoutProf As Profile = Nothing

        Dim objAsmFeat As AssemblyFeatures = Nothing
        Dim objAsmFeatExtrCOs As AssemblyFeaturesExtrudedCutouts = Nothing
        Dim objAsmFeatExtrCO As AssemblyFeaturesExtrudedCutout = Nothing

        Dim nNumScopeParts As Integer = 0
        Dim pScopeParts() As Object = Nothing
        Dim nNumProfiles As Integer = 0
        Dim pProfiles() As SolidEdgePart.Profile = Nothing
        Dim ExtentType As SolidEdgeConstants.FeaturePropertyConstants
        Dim pExtentSide As SolidEdgeConstants.FeaturePropertyConstants
        Dim profileSide As SolidEdgeConstants.FeaturePropertyConstants
        Dim pdDistance As Double = 0
        Dim pKeyPoint As Object = Nothing
        Dim pKeyPointFlags As SolidEdgeConstants.KeyPointExtentConstants
        Dim pFromSurfOrPlane As Object = Nothing
        Dim pToSurfOrPlane As Object = Nothing
        Dim SEInstallDir As DirectoryInfo

        Try
            Debug.Print("[START]-------")

            objSEApplication = Marshal.GetActiveObject("SolidEdge.Application")
            SEInstallDir = GetTrainingFolder()

            objDoc = objSEApplication.Documents.Open(SEInstallDir.FullName + "\movprt1.asm")

            ' Activate occurrences.
            objOccs = objDoc.Occurrences

            ' Get information of lever1.par.
            objOcc = objOccs.Item(5)
            objPartDoc = objOcc.OccurrenceDocument
            objModel = objPartDoc.Models.Item(1)


            ' Create referece plane from lever1.par -> ExtrudedProtrusions.
            objExtrPRs = objModel.ExtrudedProtrusions
            objExtrPR = objExtrPRs.Item(2)
            objExtrPRProf = objExtrPR.Profile
            objFaces = objExtrPR.Faces(
                SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryPlane)

            objFace = objFaces.Item(1)
            objEdge = objFace.Edges.item(1)
            objFaceRef = objDoc.CreateReference(objOcc, objFace)
            objEdgeRef = objDoc.CreateReference(objOcc, objEdge)


            ' Create assembly referce plane.
            objAsmRefPlanes = objDoc.AsmRefPlanes
            objAsmRefPlane = objAsmRefPlanes.AddParallelByDistance(objFaceRef, 0)
            objAsmRefPlane.Visible = True


            ' Create layout.
            objLayouts = objDoc.Layouts
            objLayout = objLayouts.Add(objAsmRefPlane)
            objLayoutProf = objLayout.Profile
            Try
                objLayoutProf.IncludeEdge(objEdge)
            Catch ex As Exception
                Debug.Print("[EXCEPTION]---")
                Debug.Print(" > " & ex.Message)
            End Try


            ' Set AssemblyFeatures informations
            nNumScopeParts = 1
            ReDim pScopeParts(nNumScopeParts - 1)
            pScopeParts(0) = objOcc

            nNumProfiles = 1
            ReDim pProfiles(nNumProfiles - 1)
            pProfiles(0) = objLayoutProf ' Layout

            ExtentType = SolidEdgeConstants.FeaturePropertyConstants.igFinite
            pExtentSide = SolidEdgeConstants.FeaturePropertyConstants.igLeft
            profileSide = SolidEdgeConstants.FeaturePropertyConstants.igLeft
            pdDistance = 0.01
            pKeyPointFlags = SolidEdgeConstants.KeyPointExtentConstants.igTangentNormal
            pKeyPoint = Nothing
            pFromSurfOrPlane = Nothing
            pToSurfOrPlane = Nothing
            ' Create an AssemblyFeaturesExtrudedCutout
            objAsmFeat = objDoc.AssemblyFeatures
            objAsmFeatExtrCOs = objAsmFeat.AssemblyFeaturesExtrudedCutouts
            objAsmFeatExtrCO = objAsmFeatExtrCOs.Add(
                nNumScopeParts,
                pScopeParts,
                nNumProfiles,
                pProfiles,
                ExtentType,
                pExtentSide,
                profileSide,
                pdDistance,
                pKeyPoint,
                pKeyPointFlags,
                pFromSurfOrPlane,
                pToSurfOrPlane
            )
            objDoc.UpdatePathfinder(AssemblyPathfinderUpdateConstants.seRebuild)

            Debug.Print("[END]---------")

        Catch ex As Exception
            Debug.Print("[EXCEPTION]---------")

    }
