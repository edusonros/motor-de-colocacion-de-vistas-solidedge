Imports System.IO

Public Class Form1
    Function GetTrainingFolder() As DirectoryInfo
        Dim objInstallData As SEInstallDataLib.SEInstallData = Nothing
        Dim objInstallFolder As DirectoryInfo = Nothing
        Dim objTrainingFolder As DirectoryInfo = Nothing

        Try
            objInstallData = New SEInstallDataLib.SEInstallData
            objInstallFolder = New DirectoryInfo(objInstallData.GetInstalledPath())
            objTrainingFolder = New DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"))
        Catch
        Finally
            If Not (objInstallData Is Nothing) Then
                Marshal.FinalReleaseComObject(objInstallData)
                objInstallData = Nothing
            End If
        End Try

        Return objTrainingFolder
    End Function

    Private Sub AssemblyFeaturesHoles_Add_Click(sender As Object, e As EventArgs) Handles AssemblyFeaturesHoles_Add.Click
        Dim objApplication As SolidEdgeFramework.Application = Nothing
        Dim objAssemblyDocument As SolidEdgeAssembly.AssemblyDocument = Nothing
        Dim SEInstallDir As DirectoryInfo
        Dim objOccurrences As SolidEdgeAssembly.Occurrences = Nothing
        Dim objOccurrence As SolidEdgeAssembly.Occurrence = Nothing
        Dim objLayouts As SolidEdgeAssembly.Layouts = Nothing
        Dim objLayout As SolidEdgeAssembly.Layout = Nothing
        Dim objLayoutProfile As SolidEdgePart.Profile = Nothing
        Dim objAsmFeatHoles As SolidEdgeAssembly.AssemblyFeaturesHoles = Nothing
        Dim objAsmFeatHole As SolidEdgeAssembly.AssemblyFeaturesHole = Nothing
        Dim ExtentType As SolidEdgePart.FeaturePropertyConstants = Nothing
        Dim ExtentSide As SolidEdgePart.FeaturePropertyConstants = Nothing
        Dim ProfileSide As SolidEdgePart.FeaturePropertyConstants = Nothing
        Dim objProfile As Object
        Dim arrProfile(1) As Object
        Dim pFromSurfOrPlane As Object
        Dim pKeyPoint As Object
        Dim pToSurfOrPlane As Object
        Dim pKeyPointFlags As SolidEdgePart.KeyPointExtentConstants
        Dim arrScopeParts() As Object
        Dim nParts As Short
        Dim i As Short
        Dim objHoleData As Object = Nothing
        Try
            objApplication = Marshal.GetActiveObject("SolidEdge.Application")
            SEInstallDir = GetTrainingFolder()
            objAssemblyDocument = objApplication.Documents.Open(SEInstallDir.FullName + "\carrier.asm")
            'objAssemblyDocument = objApplication.ActiveDocument
            objHoleData = objAssemblyDocument.HoleDataCollection.Add(HoleType:=SolidEdgePart.FeaturePropertyConstants.igCounterboreHole, _
                                                   HoleDiameter:=0.005, CounterboreDiameter:=0.007, _
                                                   CounterboreDepth:=0.0075)
            objAsmFeatHoles = objAssemblyDocument.AssemblyFeatures.AssemblyFeaturesHoles

            Dim objAsmRefPlanes As SolidEdgeAssembly.AsmRefPlanes
            Dim objAsmRefPlane As SolidEdgeAssembly.AsmRefPlane
            objAsmRefPlanes = objAssemblyDocument.AsmRefPlanes
            objAsmRefPlane = objAsmRefPlanes.AddParallelByDistance(objAsmRefPlanes(0), 0.02111)

            'Get the Layouts Collection
            objLayouts = objAssemblyDocument.Layouts
            'Create a new Layout
            objLayout = objLayouts.Add(AsmRefPlane:=objAsmRefPlane)

            'Get the Profile object from Layout
            objLayoutProfile = objLayout.Profile
            'Now draw a circle
            objLayoutProfile.Holes2d.Add(0.0, 0.03)

            objProfile = objLayout.Profile ' This is the same as a part profile.
            arrProfile(0) = objProfile '0 based index

            'Setup the scoped parts - use all the refrences

            nParts = objAssemblyDocument.Occurrences.Count
            ReDim arrScopeParts(nParts)
            'Has to give the parts to be modified by the feature
            For i = 1 To objAssemblyDocument.Occurrences.Count
                arrScopeParts(i - 1) = objAssemblyDocument.Occurrences.Item(i)
            Next
            'Exent type such as finite, through all , etc.
            ExtentType = SolidEdgeConstants.FeaturePropertyConstants.igThroughAll

            'Profile side.
            ExtentSide = SolidEdgeConstants.FeaturePropertyConstants.igLeft

            'Extent side - needed for finite extent type
            ProfileSide = SolidEdgeConstants.FeaturePropertyConstants.igLeft
            pKeyPoint = 0
            pFromSurfOrPlane = 0
            pToSurfOrPlane = 0

            objAsmFeatHole = objAsmFeatHoles.Add(4, arrScopeParts, 1, arrProfile, ExtentSide, objHoleData, ExtentType, 0.03, pFromSurfOrPlane, pToSurfOrPlane, pKeyPoint, pKeyPointFlags)

            'Update Path finder
            objAssemblyDocument.UpdatePathfinder(SolidEdgeAssembly.AssemblyPathfinderUpdateConstants.seRebuild)

        Catch ex As Exception
            MsgBox(ex.ToString)
            Clipboard.SetText(ex.ToString)
        End Try
    End Sub

    public DirectoryInfo GetTrainingFolder()
    {
        SEInstallDataLib.SEInstallData objInstallData = null;
        DirectoryInfo objInstallFolder = null;
        DirectoryInfo objTrainingFolder = null;

        try
        {
            objInstallData = new SEInstallDataLib.SEInstallData();
            objInstallFolder = new DirectoryInfo(objInstallData.GetInstalledPath());
            objTrainingFolder = new DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"));
        }
        catch
        {
        }
        finally
        {
            if (objInstallData != null)
            {
                Marshal.FinalReleaseComObject(objInstallData);
                objInstallData = null;
            }
        }

        return objTrainingFolder;
    }

    private void AssemblyFeaturesHoles_Add_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgeAssembly.AssemblyDocument objAssemblyDocument = null;
        DirectoryInfo SEInstallDir = null;
        SolidEdgeAssembly.Occurrences objOccurrences = null;
        SolidEdgeAssembly.Occurrence objOccurrence = null;
        SolidEdgeAssembly.Layouts objLayouts = null;
        SolidEdgeAssembly.Layout objLayout = null;
        SolidEdgePart.Profile objLayoutProfile = null;
        SolidEdgeAssembly.AssemblyFeaturesHoles objAsmFeatHoles = null;
        SolidEdgeAssembly.AssemblyFeaturesHole objAsmFeatHole = null;
        SolidEdgePart.FeaturePropertyConstants ExtentType = null;
        SolidEdgePart.FeaturePropertyConstants ExtentSide = null;
        SolidEdgePart.FeaturePropertyConstants ProfileSide = null;
        object objProfile = null;
        object[] arrProfile = new object[2];
        object pFromSurfOrPlane = null;
        object pKeyPoint = null;
        object pToSurfOrPlane = null;
        SolidEdgePart.KeyPointExtentConstants pKeyPointFlags = null;
        object[] arrScopeParts = null;
        short nParts = 0;
        short i = 0;
        object objHoleData = null;
        try
        {
            objApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            SEInstallDir = GetTrainingFolder();
            objAssemblyDocument = objApplication.Documents.Open(SEInstallDir.FullName + "\\carrier.asm");
            //objAssemblyDocument = objApplication.ActiveDocument
            objHoleData = objAssemblyDocument.HoleDataCollection.Add(HoleType:SolidEdgePart.FeaturePropertyConstants.igCounterboreHole, HoleDiameter:0.005, CounterboreDiameter:0.007, CounterboreDepth:0.0075);
            objAsmFeatHoles = objAssemblyDocument.AssemblyFeatures.AssemblyFeaturesHoles;

            SolidEdgeAssembly.AsmRefPlanes objAsmRefPlanes = null;
            SolidEdgeAssembly.AsmRefPlane objAsmRefPlane = null;
            objAsmRefPlanes = objAssemblyDocument.AsmRefPlanes;
            objAsmRefPlane = objAsmRefPlanes.AddParallelByDistance(objAsmRefPlanes(0), 0.02111);

            //Get the Layouts Collection
            objLayouts = objAssemblyDocument.Layouts;
            //Create a new Layout
            objLayout = objLayouts.Add(AsmRefPlane:objAsmRefPlane);

            //Get the Profile object from Layout
            objLayoutProfile = objLayout.Profile;
            //Now draw a circle
            objLayoutProfile.Holes2d.Add(0.0, 0.03);

            objProfile = objLayout.Profile; // This is the same as a part profile.
            arrProfile[0] = objProfile; //0 based index

            //Setup the scoped parts - use all the refrences

            nParts = (short)objAssemblyDocument.Occurrences.Count;
            arrScopeParts = new object[nParts + 1];
            //Has to give the parts to be modified by the feature
            for (i = 1; i <= objAssemblyDocument.Occurrences.Count; i++)
            {
                arrScopeParts[i - 1] = objAssemblyDocument.Occurrences.Item(i);
            }
            //Exent type such as finite, through all , etc.
            ExtentType = SolidEdgeConstants.FeaturePropertyConstants.igThroughAll;

            //Profile side.
            ExtentSide = SolidEdgeConstants.FeaturePropertyConstants.igLeft;

            //Extent side - needed for finite extent type
            ProfileSide = SolidEdgeConstants.FeaturePropertyConstants.igLeft;
            pKeyPoint = 0;
            pFromSurfOrPlane = 0;
            pToSurfOrPlane = 0;

            objAsmFeatHole = objAsmFeatHoles.Add(4, arrScopeParts, 1, arrProfile, ExtentSide, objHoleData, ExtentType, 0.03, pFromSurfOrPlane, pToSurfOrPlane, pKeyPoint, pKeyPointFlags);

            //Update Path finder
            objAssemblyDocument.UpdatePathfinder(SolidEdgeAssembly.AssemblyPathfinderUpdateConstants.seRebuild);

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
            Clipboard.SetText(ex.ToString());
        }
    }

    public Form1()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        AssemblyFeaturesHoles_Add.Click += AssemblyFeaturesHoles_Add_Click;
    }
