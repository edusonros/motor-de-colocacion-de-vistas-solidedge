using System;
using System.Runtime.InteropServices;
internal static class InitiateWorkflow
{

    public static void Main()
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgeFramework.SolidEdgeTCE objTCE = null;
        object listOfWrkFlows = null;
        string strTemp = "TCM Release Process"; //we can pass the workflow returned from GetListOfWorkFlows method
        object obj1 = null;
        string UserName = null;
        string Password = null;
        string Group = null;
        string Role = null;
        string URL = null;

        try
        {
            objApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");

            //Get the teamcenter object  
            objTCE = objApplication.SolidEdgeTCE;

            UserName = "user1";
            Password = "user1";
            Group = "Engineering";
            Role = "Designer";
            URL = "http://servername:7001/tc";

            objTCE.ValidateLogin(UserName, Password, Group, Role, URL);

            //For example I have taken file whose item revision is 011157 
            object[,] MFKAttributes = new object[3, 2];
            MFKAttributes[0, 0] = "item_id";
            MFKAttributes[0, 1] = "011157";
            MFKAttributes[1, 0] = "object_type";
            MFKAttributes[1, 1] = "SE99_P4447_02";
            MFKAttributes[2, 0] = "se99_attr1";
            MFKAttributes[2, 1] = "hhhh";

            string FileItemRevID = "A";
            object ItemTypeMFK = "SE99_P4447_02";

            //Get the list of workflows    
            objTCE.GetListOfWorkflows(MFKAttributes, ItemTypeMFK, FileItemRevID, true, listOfWrkFlows); //False returns all the worfklows.

        }
        catch (Exception ex)
        {
            Console.WriteLine(ex.Message);
        }

    }

}
