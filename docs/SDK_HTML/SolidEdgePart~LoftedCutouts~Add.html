Control the start condition, the shape of the Loft feature where it meets the first cross section, with the following options.
igNatural - There is no constraining condition enforced at the end. This is the default condition and is valid for any cross section type.
igNormal - The loft is perpendicular to the reference plane of the cross section.
For the start type of igNormal, this value provides control of the shape of the Loft at the first cross section.
Control the end condition, the shape of the loft feature where it meets the last cross section, with the following options.
igNatural - There is no constraining condition enforced at the end. This is the default condition and is valid for any cross section type.
igNormal - The loft is perpendicular to the reference plane of the cross section.
For the end type of igNormal, this value provides control of the shape of the Loft at the last cross section.
Number of guide curves.
An array containing the objects representing the path along which the cross sections are to be swept.  The length of the array is defined by the NumCurves parameter.

Adds an occurrence of the referenced object.

Private Sub Form_Load()
    Dim objApp As SolidEdgeFramework.Application
    Dim objDoc As SolidEdgePart.PartDocument
    Dim objBaseProf(1 To 3) As SolidEdgePart.Profile
    Dim objLoftProtProf1 As SolidEdgePart.Profile
    Dim objLoftProtProf2 As SolidEdgePart.Profile
    Dim objBaseCirc1 As SolidEdgeFrameworkSupport.Circle2d
    Dim objBaseCirc2 As SolidEdgeFrameworkSupport.Circle2d
    Dim objLoftProtCirc1 As SolidEdgeFrameworkSupport.Circle2d
    Dim objLoftProtCirc2 As SolidEdgeFrameworkSupport.Circle2d
    Dim objModel As SolidEdgePart.Model
    Dim objRefPln As SolidEdgePart.RefPlane
    Dim objLoftProt1 As SolidEdgePart.LoftedProtrusion
    Dim objLoftProt2 As SolidEdgePart.LoftedProtrusion
    Dim objLoftCut As SolidEdgePart.LoftedCutout
    Dim objSections(1 To 3) As Object
    Dim objSections2(1 To 3) As Object
    Dim objEdges1 As Object
    Dim objEdges2 As Object
    Dim lngStatus As Long
    Dim xOrigin As Double, yOrigin As Double
    Dim OriginArray(1 To 3) As Variant
    Dim Origin(1 To 2) As Double
    Dim SectionTypes(1 To 3) As Long
    Dim SectionTypes2(1 To 3) As Long
    Dim i As Integer
    ' Report errors
    Const PI = 3.14159265358979
    ' Create/get the application with specific settings
    On Error Resume Next
    Set objApp = GetObject(, "SolidEdge.Application")
    If Err Then
        Err.Clear
        Set objApp = CreateObject("SolidEdge.Application")
        Set objDoc = objApp.Documents.Add("SolidEdge.PartDocument")
        objApp.Visible = True
    Else
        Set objDoc = objApp.ActiveDocument
    End If
    On Error GoTo 0
    ' Draw the Base Profile
    Set objBaseProf(1) = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:= _
                                                             objDoc.RefPlanes(2))
    Set objBaseCirc1 = objBaseProf(1).Circles2d.AddByCenterRadius(x:=0, y:=0, Radius:=0.03)
    ' Get the model space origin relative to profile space.
    Call objBaseProf(1).Convert3DCoordinate(x3d:=0, y3d:=0, z3d:=0, _
                                            x2d:=xOrigin, y2d:=yOrigin)
    ' Check for the Profile Validity
    lngStatus = objBaseProf(1).End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox ("Profile not closed")
    End If
    ' Create a 2nd Profile to use as section
    Set objRefPln = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:= _
                                                           objDoc.RefPlanes(2), Distance:=0.05, NormalSide:=igRight)
    Set objBaseProf(2) = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:= _
                                                             objRefPln)
    Set objBaseCirc2 = objBaseProf(2).Circles2d.AddByCenterRadius(x:=0, y:=0, Radius:=0.05)
    ' Get the model space origin relative to profile space.
    Call objBaseProf(2).Convert3DCoordinate(x3d:=0, y3d:=0, z3d:=0, _
                                            x2d:=xOrigin, y2d:=yOrigin)
    ' Check for the Profile Validity
    lngStatus = objBaseProf(2).End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox ("Profile not closed")
    End If
    ' Get the Profies to use as the sections of the loft.
    Set objSections(1) = objBaseProf(1)
    Set objSections(2) = objBaseProf(2)
    ' Build up the origin array. Since these are both circles the values don't matter.
    Origin(1) = 0
    Origin(2) = 0
    OriginArray(1) = Origin
    OriginArray(2) = Origin
    ' Define the types of sections used.
    SectionTypes(1) = igProfileBasedCrossSection
    SectionTypes(2) = igProfileBasedCrossSection

    'Create the Base Extruded Protrusion Feature
    Set objModel = objDoc.Models.AddLoftedProtrusion(NumSections:=2, _
                                                     CrossSections:=objSections, _
                                                     CrossSectionTypes:=SectionTypes, Origins:=OriginArray, _
                                                     SegmentMaps:=0, MaterialSide:=igLeft, _
                                                     StartExtentType:=igNone, StartExtentDistance:=0, _
                                                     StartSurfaceOrRefPlane:=Nothing, EndExtentType:=igNone, _
                                                     EndExtentDistance:=0, EndSurfaceOrRefPlane:=Nothing, _
                                                     StartTangentType:=igTangent, StartTangentMagnitude:=0.5, _
                                                     EndTangentType:=igTangent, EndTangentMagnitude:=0.5)
    objBaseProf(1).Visible = False
    objBaseProf(2).Visible = False
    ' Check the status of Feature
    If objModel.LoftedProtrusions(1).Status <> igFeatureOK Then
        MsgBox ("Error in the Creation of LoftedProtrusion Feature object")
    End If
    Set objLoftProt1 = objModel.LoftedProtrusions(1)
    ' Create a 2nd LoftedProtrusion
    ' Draw the Profile
    Set objRefPln = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:= _
                                                           objDoc.RefPlanes(2), Distance:=0.05, NormalSide:=igRight)
    Set objLoftProtProf1 = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:= _
                                                               objRefPln)
    Set objLoftProtCirc1 = objLoftProtProf1.Circles2d.AddByCenterRadius(x:=0, y:=0, Radius:=0.03)
    ' Get the model space origin relative to profile space.
    Call objLoftProtProf1.Convert3DCoordinate(x3d:=0, y3d:=0, z3d:=0, _
                                              x2d:=xOrigin, y2d:=yOrigin)
    ' Save the coordinates to define the start point for this profile.
    Origin(1) = xOrigin
    Origin(2) = yOrigin
    OriginArray(1) = Origin
    ' Check for the Profile Validity
    lngStatus = objLoftProtProf1.End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox ("Profile not closed")
    End If
    ' Create a 2nd Profile to use as section
    Set objRefPln = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:= _
                                                           objDoc.RefPlanes(2), Distance:=0.1, NormalSide:=igRight)
    Set objLoftProtProf2 = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:= _
                                                               objRefPln)
    Set objLoftProtCirc2 = objLoftProtProf2.Circles2d.AddByCenterRadius(x:=0, y:=0, Radius:=0.01)
    ' Get the model space origin relative to profile space.
    Call objLoftProtProf2.Convert3DCoordinate(x3d:=0, y3d:=0, z3d:=0, _
                                              x2d:=xOrigin, y2d:=yOrigin)
    ' Save the coordinates to define the start point for this profile.
    Origin(1) = xOrigin
    Origin(2) = yOrigin
    OriginArray(2) = Origin
    ' Check for the Profile Validity
    lngStatus = objLoftProtProf2.End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox ("Profile not closed")
    End If
    ' Get the Profies to use as the sections of the loft.
    Set objSections(1) = objLoftProtProf1
    Set objSections(2) = objLoftProtProf2
    ' Build up the origin array. Since these are both circles the values don't matter.
    Origin(1) = 0
    Origin(2) = 0
    OriginArray(1) = Origin
    OriginArray(2) = Origin
    ' Define the types of sections used.
    SectionTypes(1) = igProfileBasedCrossSection
    SectionTypes(2) = igProfileBasedCrossSection

    'Create the LoftedProtrusion Feature
    Set objLoftProt2 = objModel.LoftedProtrusions.Add(NumSections:=2, _
                                                      CrossSections:=objSections, _
                                                      CrossSectionTypes:=SectionTypes, Origins:=OriginArray, _
                                                      SegmentMaps:=0, MaterialSide:=igLeft, _
                                                      StartExtentType:=igNone, StartExtentDistance:=0, _
                                                      StartSurfaceOrRefPlane:=Nothing, EndExtentType:=igNone, _
                                                      EndExtentDistance:=0, EndSurfaceOrRefPlane:=Nothing, _
                                                      StartTangentType:=igTangent, StartTangentMagnitude:=0.5, _
                                                      EndTangentType:=igTangent, EndTangentMagnitude:=0.5)
    objLoftProtProf1.Visible = False
    objLoftProtProf2.Visible = False
    ' Check the status of Feature
    If objLoftProt2.Status <> igFeatureOK Then
        MsgBox ("Error in the Creation of LoftedProtrusion Feature object")
    End If
    ' Create the LoftedCutout
    Set objEdges1 = objLoftProt1.Edges(igQueryAll)
    Set objSections2(1) = objEdges1(1)
    Set objEdges2 = objLoftProt2.Edges(igQueryAll)
    Set objSections2(2) = objEdges2(2)
    ' Build up the origin array. Since these are both circles the values don't matter.
    Origin(1) = 0
    Origin(2) = 0
    OriginArray(1) = Origin
    OriginArray(2) = Origin
    ' Define the types of sections used.
    SectionTypes2(1) = igEdgeBasedCrossSection
    SectionTypes2(2) = igEdgeBasedCrossSection
    ' Build up an array of the origin positions.
    Set objLoftCut = objModel.LoftedCutouts.Add(NumSections:=2, CrossSections:=objSections2, _
                                                CrossSectionTypes:=SectionTypes2, Origins:=OriginArray, _
                                                SegmentMaps:=0, MaterialSide:=igLeft, _
                                                StartExtentType:=igNone, StartExtentDistance:=0.01, _
                                                StartSurfaceOrRefPlane:=Nothing, EndExtentType:=igNone, _
                                                EndExtentDistance:=0, EndSurfaceOrRefPlane:=Nothing, _
                                                StartTangentType:=igTangent, StartTangentMagnitude:=0.5, _
                                                EndTangentType:=igTangent, EndTangentMagnitude:=0.5)

    ' USER DISPLAY
    ' Release objects
    Set objApp = Nothing
    Set objDoc = Nothing
    Set objBaseProf(1) = Nothing
    Set objBaseProf(2) = Nothing
    Set objLoftProtProf1 = Nothing
    Set objLoftProtProf2 = Nothing
    Set objLoftProtCirc1 = Nothing
    Set objLoftProtCirc2 = Nothing
    Set objBaseCirc1 = Nothing
    Set objBaseCirc2 = Nothing
    Set objModel = Nothing
    Set objLoftProt1 = Nothing
    Set objLoftProt2 = Nothing
    Set objLoftCut = Nothing
    Set objRefPln = Nothing
    Set objSections(1) = Nothing
    Set objSections(2) = Nothing
    Set objSections2(1) = Nothing
    Set objSections2(1) = Nothing
    Set objEdges1 = Nothing
    Set objEdges2 = Nothing
End Sub
