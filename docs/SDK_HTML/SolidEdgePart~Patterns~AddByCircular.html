Public Class Form1
    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApp As SolidEdgeFramework.Application
        Dim objDoc As SolidEdgePart.PartDocument
        Dim objPattern As SolidEdgePart.Pattern
        Dim objFeatArray(5) As Object
        Dim objFeatArrayOut(5) As Object
        Dim objPatternPlane As SolidEdgePart.RefPlane
        Dim OriginPos(3) As Double
        Dim AxisPos(3) As Double
        Dim objSuppressRegionsArray(5) As Object
        Dim objHoles As SolidEdgePart.Holes = Nothing

        Try

            objSEApp = GetObject(, "SolidEdge.Application")
            'Create a part document having a disc and a hole in it. 
            'Make sure that it is created using synchronous mode since add by circular is only supported in sync mode
            objDoc = objSEApp.ActiveDocument
            objHoles = objDoc.Models.Item(1).Holes
            objFeatArray(0) = objHoles.Item(1)

            objPatternPlane = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:=objDoc.RefPlanes.Item(3),
                                    Distance:=0.0, NormalSide:=SolidEdgeConstants.ReferenceElementConstants.igNormalSide, Local:=True)

            AxisPos(0) = 0.0
            AxisPos(1) = 0
            AxisPos(2) = 0.0

            objPattern = objDoc.Models.Item(1).Patterns.AddByCircular(1, objFeatArray,
                        objPatternPlane, 5, 0.0, AxisPos,
                        SolidEdgeConstants.PatternOffsetTypeConstants.sePatternFitOffset,
                        SolidEdgeConstants.PatternCurveAnchorSideConstants.sePatternCurveRightSide, False)
            If objPattern Is Nothing Then
                MsgBox("Could not create pattern by AddByCircular ")
            End If

        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System;
using System.Windows.Forms;

public class Form1
{
    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objSEApp = null;
        SolidEdgePart.PartDocument objDoc = null;
        SolidEdgePart.Pattern objPattern = null;
        object[] objFeatArray = new object[6];
        object[] objFeatArrayOut = new object[6];
        SolidEdgePart.RefPlane objPatternPlane = null;
        double[] OriginPos = new double[4];
        double[] AxisPos = new double[4];
        object[] objSuppressRegionsArray = new object[6];
        SolidEdgePart.Holes objHoles = null;

        try
        {

            objSEApp = Microsoft.VisualBasic.Interaction.GetObject(null, "SolidEdge.Application");
            //Create a part document having a disc and a hole in it. 
            //Make sure that it is created using synchronous mode since add by circular is only supported in sync mode
            objDoc = objSEApp.ActiveDocument;
            objHoles = objDoc.Models.Item(1).Holes;
            objFeatArray[0] = objHoles.Item(1);

            objPatternPlane = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:objDoc.RefPlanes.Item(3), Distance:0.0, NormalSide:SolidEdgeConstants.ReferenceElementConstants.igNormalSide, Local:true);

            AxisPos[0] = 0.0;
            AxisPos[1] = 0;
            AxisPos[2] = 0.0;

            objPattern = objDoc.Models.Item(1).Patterns.AddByCircular(1, objFeatArray, objPatternPlane, 5, 0.0, AxisPos, SolidEdgeConstants.PatternOffsetTypeConstants.sePatternFitOffset, SolidEdgeConstants.PatternCurveAnchorSideConstants.sePatternCurveRightSide, false);
            if (objPattern == null)
            {
                MessageBox.Show("Could not create pattern by AddByCircular ");
            }

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }

    public Form1()
    {
        SubscribeToEvents();
    }


    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
