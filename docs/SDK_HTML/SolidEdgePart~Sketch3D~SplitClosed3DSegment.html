    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objApp As SolidEdgeFramework.Application = Nothing

        Dim objDoc As SolidEdgePart.PartDocument = Nothing

        Dim objSketches3D As SolidEdgePart.Sketches3D = Nothing
        Dim objSketch3D As SolidEdgePart.Sketch3D = Nothing

        Dim objCircles As SolidEdgePart.Ellipses3D = Nothing
        Dim objCircle1 As SolidEdgePart.Ellipse3D = Nothing
        Dim objCircle2 As SolidEdgePart.Ellipse3D = Nothing
        Dim objSplitSeg3D As SolidEdgePart.Ellipse3D = Nothing
        Dim objArcSeg3D As SolidEdgePart.Arc3D = Nothing
        Dim splitElementEndPos(0 To 2) As Double

        Dim objInstanceRefArray(1) As Object
        Dim IsCircle As Boolean
        Dim iCount As Integer

        ' Report errors
        Try
            ' Create/get the application with specific settings
            objApp = GetObject(, "SolidEdge.Application")

            ' open the document mentioned in the doc files.
            objDoc = objApp.Documents.Add("SolidEdge.PartDocument")

            objSketches3D = objDoc.Sketches3D
            ' Now add new sketch
            objSketch3D = objSketches3D.Add()
            objCircles = objSketch3D.Ellipses3D

            'Create  the Circle by radius,normal and center     
            objCircle1 = objCircles.AddByCenterRadiusNormal(0, 0.002, 0.005, 1, 0, 0, 0.04)

            'Try for some properties of Arc
            objCircle1 = objCircles.Item(1)

            IsCircle = objCircle1.IsCircle()
            iCount = objCircles.Count   

            objInstanceRefArray(0) = Nothing
            objInstanceRefArray(1) = Nothing

            objSketch3D.SplitClosed3DSegment(objCircle1, 0.0, 0.032236469153492965, 0.031186942034721592, 0.0, -0.026970748722462329, -0.022580712798257182, objInstanceRefArray)
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try

    End Sub
End Class
