    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApplication As SolidEdgeFramework.Application = Nothing
        Dim objAsmDoc As SolidEdgeAssembly.AssemblyDocument = Nothing
        Dim objAsmOcc As SolidEdgeAssembly.Occurrence = Nothing
        Dim objCloneOcc As SolidEdgeAssembly.Occurrence = Nothing
        Dim objRefOcc As SolidEdgeAssembly.Occurrence = Nothing
        Dim objComponenetsToClone As Array = Array.CreateInstance(GetType(Object), 1)
        Dim objFaces As Array = Array.CreateInstance(GetType(Object), 1)
        Dim objCloneEnv As Array = Array.CreateInstance(GetType(Object), 1)
        Dim objPartDoc As SolidEdgePart.PartDocument = Nothing
        Dim objPartBody As SolidEdgeGeometry.Body = Nothing
        Dim ErrorStatus As Long
        Dim objTrainingFolder As DirectoryInfo
        Dim CloneComponentType As Object = SolidEdgeConstants.CloneComponentOptions.seDoNotCreateRelationships
        Try


         objSEApplication = Marshal.GetActiveObject("SolidEdge.Application")
            'Get Training Folder
            objTrainingFolder = GetTrainingFolder()
            'Open carrier.asm
            objSEApplication.Documents.Open(Path.Combine(objTrainingFolder.FullName, "seva04.asm"))
            objAsmDoc = objSEApplication.ActiveDocument

            ' collect all input data for clone command
            objAsmOcc = objAsmDoc.Occurrences.Item(2)
            Dim occname As String = objAsmOcc.Name

            objComponenetsToClone.SetValue(objAsmOcc, 0)

            ' get reference geometry
            objRefOcc = objAsmDoc.Occurrences.Item(1)
            objPartDoc = objRefOcc.OccurrenceDocument
            objPartBody = objPartDoc.Models.Item(1).Body
            objFaces.SetValue(objAsmDoc.CreateReference(objRefOcc, objPartBody.GetEntityByID(228)), 0)


            ' get clone environment
            objCloneOcc = objAsmDoc.Occurrences.Item(1)
            Dim occname1 As String = objCloneOcc.Name

            objCloneEnv.SetValue(objCloneOcc, 0)
            objAsmDoc.CreateCloneComponents(objComponenetsToClone, _
                                            objFaces, _
                                            objCloneEnv, _
                                            SolidEdgeConstants.CloneComponentOptions.seDoNotCreateRelationships, _
                                            True, _
                                            SolidEdgeConstants.CloneMatchTypeOptions.CloneMatchTypeAutomatic, _
                                            ErrorStatus)

        Catch ex As Exception
            MsgBox(ex.ToString)
            If Not objAsmDoc Is Nothing Then
                Call objAsmDoc.Close()
            End If
        Finally
          
        End Try
    End Sub
    Function GetTrainingFolder() As DirectoryInfo
        Dim objInstallData As SEInstallDataLib.SEInstallData = Nothing
        Dim objInstallFolder As DirectoryInfo = Nothing
        Dim objTrainingFolder As DirectoryInfo = Nothing

        Try
            objInstallData = New SEInstallDataLib.SEInstallData
            objInstallFolder = New DirectoryInfo(objInstallData.GetInstalledPath())
            objTrainingFolder = New DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"))
        Catch
        Finally
            If Not (objInstallData Is Nothing) Then
                Marshal.FinalReleaseComObject(objInstallData)
                objInstallData = Nothing
            End If
        End Try

        Return objTrainingFolder
    End Function

        Array objFaces = (Array[])Array.CreateInstance(typeof(object), 1);

        SolidEdgeGeometry.Body objPartBody = null;

        DirectoryInfo objTrainingFolder = null;
        object CloneComponentType = SolidEdgeConstants.CloneComponentOptions.seDoNotCreateRelationships;
        try
        {


         objSEApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            //Get Training Folder
            objTrainingFolder = GetTrainingFolder();
            //Open carrier.asm
            objSEApplication.Documents.Open(Path.Combine(objTrainingFolder.FullName, "seva04.asm"));

            objAsmOcc = objAsmDoc.Occurrences.Item(2);

    public DirectoryInfo GetTrainingFolder()

        }
