Imports System.Runtime.InteropServices

Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
       
        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objPatterns As SolidEdgePart.Patterns = Nothing
        Dim objPattrn As SolidEdgePart.Pattern = Nothing

        Dim objFeatArray(5) As Object
        Dim NumOfFeats As Integer
        Dim objCoOrdinateSystem As SolidEdgePart.CoordinateSystem
        Dim objCoOrdinateSystems As SolidEdgePart.CoordinateSystems

        Dim objPatternMethod As SolidEdgeConstants.PatternOffsetTypeConstants = SolidEdgeConstants.PatternOffsetTypeConstants.sePatternFixedOffset
        Dim objEnumRefTypeForFromPointConstant As SolidEdgeConstants.ReferencePointTypeEnumForFromPointOption = SolidEdgeConstants.ReferencePointTypeEnumForFromPointOption.sePatternReferenceTypeFromCoOrdinateSystem
        Dim objEnumRefTypeForToPointConstant As SolidEdgeConstants.ReferencePointTypeEnumForToPointOption = SolidEdgeConstants.ReferencePointTypeEnumForToPointOption.sePatternReferenceTypeToExcelFirstRow
        Dim objPatternCurveAnchorSideDirn As SolidEdgePart.PatternCurveAnchorSideConstants = SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveLeftSide
        Dim objPatternTypeConstant As SolidEdgePart.PatternTypeConstants = SolidEdgePart.PatternTypeConstants.seSmartPattern
        Dim objAngleUnitTypeConstant As SolidEdgeConstants.UnitOfMeasureAngleReadoutConstants = SolidEdgeConstants.UnitOfMeasureAngleReadoutConstants.seAngleDegree
        Dim objDistanceUnitTypeConstant As SolidEdgeConstants.UnitOfMeasureLengthReadoutConstants = SolidEdgeConstants.UnitOfMeasureLengthReadoutConstants.seLengthMillimeter

        Dim numFeats As Integer = 0
        Dim angleUnit As SolidEdgeConstants.UnitOfMeasureAngleReadoutConstants
        Dim distcUnit As SolidEdgeConstants.UnitOfMeasureLengthReadoutConstants
        Dim BIsLink As Boolean
        Dim objFeatArr(5) As Object
        Dim objGetCoOrdinateSys As SolidEdgePart.CoordinateSystem
        Dim objPatternMetd As SolidEdgeConstants.PatternOffsetTypeConstants
        Dim objEnumRefPtFromOption As SolidEdgeConstants.ReferencePointTypeEnumForFromPointOption
        Dim objEnumRefPtToOption As SolidEdgeConstants.ReferencePointTypeEnumForToPointOption
        Dim objOutFilePath As String = "Test"
        Dim suppDirn As SolidEdgeConstants.PatternCurveAnchorSideConstants

        Dim obj2 As New Object
        Dim obj3 As New Object

        Dim Description As String
        Dim objFeatArrayOut(5) As Object
        Dim Status As SolidEdgeConstants.FeatureStatusConstants
        Dim NumCount As Integer
        Dim NumOcc As Integer
        Dim RefOccIndex As Integer
        Dim Transforms As Object
        Dim Name As String
        Dim bSuccess As Boolean
        Dim featureType As SolidEdgePart.FeatureTypeConstants
        Dim x1 As Double
        Dim y1 As Double
        Dim z1 As Double
        Dim x2 As Double
        Dim y2 As Double
        Dim z2 As Double
        Dim objParent As Object
        Dim objRetFaces As SolidEdgeGeometry.Faces
        Dim objRetEdges As SolidEdgeGeometry.Edges
        Dim objFeatTopolConst As SolidEdgeGeometry.FeatureTopologyQueryTypeConstants = SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryAll
        Dim numCountFaces As Integer
        Dim numCountEdges As Integer
        Dim strExcelPath As String = "C:\Temp\Book1.xlsx"
        Try
            ' Create/get the application with specific settings
            objApp = Marshal.GetActiveObject("SolidEdge.Application")

            ' Create a part document with simple box to use this sample.
            objDoc = objApp.ActiveDocument

            objFeatArray(0) = objDoc.Models.Item(1).ExtrudedProtrusions.Item(1)

            objCoOrdinateSystems = objDoc.CoordinateSystems
            objCoOrdinateSystem = objCoOrdinateSystems.Item(1)

            objPatterns = objDoc.Models.Item(1).Patterns
            objPattrn = objPatterns.AddPatternByTable(1, objFeatArray, objCoOrdinateSystem, objPatternTypeConstant, objPatternMethod,
                                                      objEnumRefTypeForFromPointConstant, Nothing, objAngleUnitTypeConstant, objDistanceUnitTypeConstant,
                                                      strExcelPath, objPatternCurveAnchorSideDirn, True, objEnumRefTypeForToPointConstant, Nothing)


            objGetCoOrdinateSys = objDoc.CoordinateSystems.Item(1)
            objPattrn = objDoc.Models.Item(1).Patterns.Item(1)

            objPattrn.GetPatternByTableData(numFeats, objFeatArr, objGetCoOrdinateSys, objPatternMetd, objEnumRefPtFromOption, obj2,
                                        angleUnit, distcUnit, objOutFilePath, suppDirn, BIsLink, objEnumRefPtToOption, obj3)


            objFeatArray(1) = objDoc.Models.Item(1).ExtrudedProtrusions.Item(2)
            objPattrn.SetPatternByTableData(2, objFeatArray, objGetCoOrdinateSys, objPatternMetd, objEnumRefPtFromOption, Nothing, angleUnit, distcUnit,
                                            objOutFilePath, suppDirn, objEnumRefPtToOption, Nothing)

            objParent = objPattrn.Parent()


            Description = ""
            Status = objPattrn.Status(Description)


            Call objPattrn.GetInputFeatures(objFeatArrayOut)


            NumCount = objDoc.Models.Item(1).Patterns.Count()


            objPattrn = objDoc.Models.Item(1).Patterns.Item(1)

            NumCount = objPattrn.NumberOfInputFeatures()


            NumOcc = objPattrn.NumberOfOccurrences()


            RefOccIndex = objPattrn.ReferenceOccurrence()


            Transforms = objPattrn.Transforms()


            objPattrn.Range(x1, y1, z1, x2, y2, z2)


            Name = objPattrn.Name()

            featureType = objPattrn.Type


            objRetFaces = objPattrn.OccurrenceFaces(2, objFeatTopolConst)

            numCountFaces = objRetFaces.Count()

            objRetEdges = objPattrn.OccurrenceEdges(2, objFeatTopolConst)

            numCountEdges = objRetEdges.Count()
           
            bSuccess = objPattrn.Suppressed(2)
        Catch ex As Exception
            MsgBox(ex.ToString)

        End Try



    End Sub
End Class
