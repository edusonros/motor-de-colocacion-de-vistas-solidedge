    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objVars As SolidEdgeFramework.Variables = Nothing
        Dim objvarList As SolidEdgeFramework.VariableList = Nothing
        Dim sComment As String

        Try
            objApp = Marshal.GetActiveObject("SolidEdge.Application")
            objDoc = objApp.ActiveDocument
            objVars = objDoc.Variables
            objvarList = objVars.Query("*")
            For i = 1 To objvarList.Count
                Dim obj As Object = objvarList.Item(i)
                If TypeOf obj Is SolidEdgeFramework.variable Then
                    Dim objVar As SolidEdgeFramework.variable = obj
                    sComment = obj.GetComment
                    objVar = Nothing
                ElseIf TypeOf obj Is SolidEdgeFrameworkSupport.Dimension Then
                    Dim objDim As SolidEdgeFrameworkSupport.Dimension = obj
                    sComment = objDim.GetComment
                    objDim = Nothing
                End If
                Marshal.FinalReleaseComObject(obj)
                obj = Nothing
            Next i

        Catch ex As Exception
            MsgBox(ex.ToString)
        Finally
            If objvarList IsNot Nothing Then
                Marshal.ReleaseComObject(objvarList)
                objvarList = Nothing
            End If
            If objVars IsNot Nothing Then
                Marshal.ReleaseComObject(objVars)
                objVars = Nothing
            End If
            If objDoc IsNot Nothing Then
                Marshal.ReleaseComObject(objDoc)
                objDoc = Nothing
            End If
            If objApp IsNot Nothing Then
                Marshal.ReleaseComObject(objApp)
                objApp = Nothing
            End If
        End Try

    End Sub

using System.Windows.Forms;
using System.Runtime.InteropServices;

public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApp = null;
        SolidEdgePart.PartDocument objDoc = null;
        SolidEdgeFramework.Variables objVars = null;
        SolidEdgeFramework.VariableList objvarList = null;
        string sComment = null;

        try
        {
            objApp = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            objDoc = objApp.ActiveDocument;
            objVars = objDoc.Variables;
            objvarList = objVars.Query("*");
            for (var i = 1; i <= objvarList.Count; i++)
            {
                dynamic obj = objvarList.Item(i);
                if (obj is SolidEdgeFramework.variable)
                {
                    SolidEdgeFramework.variable objVar = (SolidEdgeFramework.variable)obj;
                    sComment = obj.GetComment;
                    objVar = null;
                }
                else if (obj is SolidEdgeFrameworkSupport.Dimension)
                {
                    SolidEdgeFrameworkSupport.Dimension objDim = (SolidEdgeFrameworkSupport.Dimension)obj;
                    sComment = objDim.GetComment;
                    objDim = null;
                }
                Marshal.FinalReleaseComObject(obj);
                obj = null;
            }

        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
        finally
        {
            if (objvarList != null)
            {
                Marshal.ReleaseComObject(objvarList);
                objvarList = null;
            }
            if (objVars != null)
            {
                Marshal.ReleaseComObject(objVars);
                objVars = null;
            }
            if (objDoc != null)
            {
                Marshal.ReleaseComObject(objDoc);
                objDoc = null;
            }
            if (objApp != null)
            {
                Marshal.ReleaseComObject(objApp);
                objApp = null;
            }
        }

    }

    public Form1()
    {
        SubscribeToEvents();
    }

    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
