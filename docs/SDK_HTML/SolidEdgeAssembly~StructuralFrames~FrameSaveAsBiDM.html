Description
Saves structuralframes in BiDM
Syntax
Visual Basic
Public Function FrameSaveAsBiDM( _
   ByVal FilePath As String, _
   ByVal DocumentNumber As String, _
   ByVal Revision As String, _
   ByVal Title As String _
) As String
Parameters
FilePath
File path where file is to be saved.
DocumentNumber
Unique Document number to generate file name using revision
Revision
Revision Number as string
Title
Optional – Save the Provided Title property to newly created file.
Remarks
• This method will do SaveAs of frame component when BIDM mode is ON and option “Automatically name files via Document Number and Revison “is ON.
• This method generates the file name based on provided document number and Revison.
• If user has not provided the Revision number then it will take default revision.
• If user has not provided the Document number then this API fail and will not return new generated file name.
• User needs to provide location to save the new file via filepath parameter.

using System;

public class Form1
{
    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApp = null;
        SolidEdgeAssembly.AssemblyDocument objAsmDoc = null;
        SolidEdgeAssembly.StructuralFrames objFrames = null;
        SolidEdgeAssembly.StructuralFrame objFrame = null;

        try
        {
            objApp = Microsoft.VisualBasic.Interaction.GetObject(null, "SolidEdge.Application");
            if ((objApp == null))
            {
                throw new Exception("Failed to get application object");
            }

            // Open test file. 
            objAsmDoc = objApp.Documents.Open("C:\\SEWork\\00000055_A.asm");
            if ((objAsmDoc == null))
            {
                throw new Exception("Failed to get Assembly Document object");
            }

            objFrames = objAsmDoc.StructuralFrames;

            //Call FrameSaveAs API for frame components 
            string NewfileName = null;
            string prefix = null;
            int result = 0;
            object[] sedocList = null;
            result = objApp.GetNextDocumentNumbers(1, prefix, sedocList);
            if (result != 0)
            {
                throw new Exception();
            }
            string BiDMFolderName = "C:\\SEWork";

            NewfileName = objFrames.FrameSaveAsBiDM(BiDMFolderName, sedocList[0].ToString(), "", "");
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.Message);
        }

    }
}
