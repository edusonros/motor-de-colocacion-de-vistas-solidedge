    Sub Main()
        Dim objSEApp As SolidEdgeFramework.Application = Nothing
        Dim objConstraint As SolidEdgePart.FEAConstraint = Nothing
        Dim objConstraintsProxiesArray(1) As Object
        Dim eConstraintType As SolidEdgePart.FEAConstraintTypeEnum_Auto
        Dim eConstraintDirection As SolidEdgePart.LoadSymbDirOptsEnum_Auto ' FEAConstraintTypeEnum_Auto
        Dim dConstraintDirX As Double = Nothing
        Dim dConstraintDirY As Double = Nothing
        Dim dConstraintDirZ As Double = Nothing
        Dim dConstraintPosX As Double = Nothing
        Dim dConstraintPosY As Double = Nothing
        Dim dConstraintPosZ As Double = Nothing
        Dim ColorConstraintSymbol As UInt32 = Nothing
        Dim dSpacingFactor2 As Double = Nothing
        Dim dSizeFactor2 As Double = Nothing
        Dim dSteeringWheelLen2 As Double = Nothing
        Dim dwConstraintProps As UInt32 = Nothing
        Dim dConstraintSymSpacing As Double = Nothing
        Dim dConstraintSymSize As Double = Nothing
        Dim arrStudyGeomeProxies(2) As Object
        Dim objDocument As SolidEdgePart.PartDocument = Nothing
        Dim objstudy As SolidEdgePart.Study = Nothing
        Dim objConstraintOwner As SolidEdgePart.ConstraintOwner = Nothing
        Dim objCS As SolidEdgePart.CoordinateSystem = Nothing
        Dim objstudyOwner As SolidEdgePart.StudyOwner = Nothing

        Try
            objSEApp = Marshal.GetActiveObject("SolidEdge.Application")
            objstudy = Nothing

            objDocument = objSEApp.ActiveDocument '.Documents.Item(1) '.Open(Openfilename)
            objCS = objDocument.CoordinateSystems.Item(1)
            objstudyOwner = objDocument.StudyOwner

            objstudy = objstudyOwner.Item(1) 'working on study 2

            objstudy.GetConstraintOwner(objConstraintOwner)
            objConstraint = objConstraintOwner.Item(1)


            'getting all info and set the same 
            objConstraint.GetGeometries(arrStudyGeomeProxies)
            objConstraint.GetConstraintOptions(eConstraintType, eConstraintDirection, dConstraintDirX, dConstraintDirY, dConstraintDirZ, dConstraintPosX, dConstraintPosY, dConstraintPosZ, ColorConstraintSymbol, _
                                    dSpacingFactor2, dSizeFactor2, dSteeringWheelLen2, dwConstraintProps, dConstraintSymSpacing, dConstraintSymSize)
            objConstraint.GetConstraintType(eConstraintType)
            eConstraintType = SolidEdgePart.FEAConstraintTypeEnum_Auto.eCnstrTypeFixed_Auto

            objConstraintOwner.AddConstraint(arrStudyGeomeProxies, _
                                                 eConstraintType, _
                                                 eConstraintDirection, _
                                                 dConstraintDirX, _
                                                 dConstraintDirY, _
                                                 dConstraintDirZ, _
                                                 dConstraintPosX, _
                                                 dConstraintPosY, _
                                                 dConstraintPosZ, _
                                                 ColorConstraintSymbol, _
                                                 dSpacingFactor2, _
                                                 dSizeFactor2, _
                                                 dSteeringWheelLen2, _
                                                 dwConstraintProps, _
                                                 dConstraintSymSpacing, _
                                                 dConstraintSymSize, _
                                                 objConstraint)
            ' Constraint the model in X translation, Y translation and Z rotation.
            objConstraint.SetDegreesOfFreedom(True, True, False, False, False, True)

            Dim bXTrans As Boolean = False
            Dim bYTrans As Boolean = False
            Dim bZTrans As Boolean = False
            Dim bXRot As Boolean = False
            Dim bYRot As Boolean = False
            Dim bZRot As Boolean = False

            ' Determine which DOF are constrained.
            objConstraint.GetDegreesOfFreedom(bXTrans, bYTrans, bZTrans, bXRot, bYRot, bZRot)

            If bXTrans = True Then
                MsgBox("Translation in X")
            End If

            If bYTrans = True Then
                MsgBox("Translation in Y")
            End If

            If bZRot = True Then
                MsgBox("Rotation in Z")
            End If


        Catch ex As Exception
            Debug.Print(ex.ToString)
        End Try
    End Sub

using System.Windows.Forms;
using System.Runtime.InteropServices;
internal static class Module1
{

    public static void Main()
    {
        SolidEdgeFramework.Application objSEApp = null;
        SolidEdgePart.FEAConstraint objConstraint = null;
        object[] objConstraintsProxiesArray = new object[2];
        SolidEdgePart.FEAConstraintTypeEnum_Auto eConstraintType = null;
        SolidEdgePart.LoadSymbDirOptsEnum_Auto eConstraintDirection = null; // FEAConstraintTypeEnum_Auto
        double dConstraintDirX = 0;
        double dConstraintDirY = 0;
        double dConstraintDirZ = 0;
        double dConstraintPosX = 0;
        double dConstraintPosY = 0;
        double dConstraintPosZ = 0;
        UInt32 ColorConstraintSymbol = 0;
        double dSpacingFactor2 = 0;
        double dSizeFactor2 = 0;
        double dSteeringWheelLen2 = 0;
        UInt32 dwConstraintProps = 0;
        double dConstraintSymSpacing = 0;
        double dConstraintSymSize = 0;
        object[] arrStudyGeomeProxies = new object[3];
        SolidEdgePart.PartDocument objDocument = null;
        SolidEdgePart.Study objstudy = null;
        SolidEdgePart.ConstraintOwner objConstraintOwner = null;
        SolidEdgePart.CoordinateSystem objCS = null;
        SolidEdgePart.StudyOwner objstudyOwner = null;

        try
        {
            objSEApp = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            objstudy = null;

            objDocument = objSEApp.ActiveDocument; //.Documents.Item(1) '.Open(Openfilename)
            objCS = objDocument.CoordinateSystems.Item(1);
            objstudyOwner = objDocument.StudyOwner;

            objstudy = objstudyOwner.Item(1); //working on study 2

            objstudy.GetConstraintOwner(objConstraintOwner);
            objConstraint = objConstraintOwner.Item(1);


            //getting all info and set the same 
            objConstraint.GetGeometries(arrStudyGeomeProxies);
            objConstraint.GetConstraintOptions(eConstraintType, eConstraintDirection, dConstraintDirX, dConstraintDirY, dConstraintDirZ, dConstraintPosX, dConstraintPosY, dConstraintPosZ, ColorConstraintSymbol, dSpacingFactor2, dSizeFactor2, dSteeringWheelLen2, dwConstraintProps, dConstraintSymSpacing, dConstraintSymSize);
            objConstraint.GetConstraintType(eConstraintType);
            eConstraintType = SolidEdgePart.FEAConstraintTypeEnum_Auto.eCnstrTypeFixed_Auto;

            objConstraintOwner.AddConstraint(arrStudyGeomeProxies, eConstraintType, eConstraintDirection, dConstraintDirX, dConstraintDirY, dConstraintDirZ, dConstraintPosX, dConstraintPosY, dConstraintPosZ, ColorConstraintSymbol, dSpacingFactor2, dSizeFactor2, dSteeringWheelLen2, dwConstraintProps, dConstraintSymSpacing, dConstraintSymSize, objConstraint);
            // Constraint the model in X translation, Y translation and Z rotation.
            objConstraint.SetDegreesOfFreedom(true, true, false, false, false, true);

            bool bXTrans = false;
            bool bYTrans = false;
            bool bZTrans = false;
            bool bXRot = false;
            bool bYRot = false;
            bool bZRot = false;

            // Determine which DOF are constrained.
            objConstraint.GetDegreesOfFreedom(bXTrans, bYTrans, bZTrans, bXRot, bYRot, bZRot);

            if (bXTrans == true)
            {
                MessageBox.Show("Translation in X");
            }

            if (bYTrans == true)
            {
                MessageBox.Show("Translation in Y");
            }

            if (bZRot == true)
            {
                MessageBox.Show("Rotation in Z");
            }


        }
        catch (Exception ex)
        {
            Debug.Print(ex.ToString());
        }
    }

}
