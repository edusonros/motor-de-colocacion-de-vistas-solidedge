    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objModel As SolidEdgePart.Model = Nothing
        Dim objHoleGeom As SolidEdgePart.HoleGeometry = Nothing
        Dim objHoleGeoms As SolidEdgePart.HoleGeometries = Nothing
        Dim objHoleDataColl As SolidEdgePart.HoleDataCollection = Nothing
        Dim objHoleData As SolidEdgePart.HoleData = Nothing
        Dim objTmpHoleData As SolidEdgePart.HoleData = Nothing
        Dim objHoleProf As SolidEdgePart.Profile = Nothing
        Dim objRefPln As SolidEdgePart.RefPlane = Nothing
        Dim objHole As SolidEdgePart.Hole = Nothing
        Dim lngStatus As Long = 0
        Dim ii As Long = 0

        Try
            objApp = Marshal.GetActiveObject("SolidEdge.Application")
            'sample assumes a file exists containing a Part Copy of a file that has at least one Hole feature.
            objDoc = objApp.ActiveDocument

            objHoleDataColl = objDoc.HoleDataCollection

            objModel = objDoc.Models.Item(1)

            objHoleGeoms = objModel.HoleGeometries

            objHoleGeom = objHoleGeoms.Item(1)

            objHoleData = objHoleDataColl.Copy(objHoleGeom.HoleData)

            objRefPln = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:=objDoc.RefPlanes.Item(3),
                                                               Distance:=0.0#,
                                                               NormalSide:=SolidEdgePart.FeaturePropertyConstants.igRight)

            objHoleProf = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objRefPln)

            Call objHoleProf.Holes2d.Add(XCenter:=0.04, YCenter:=0.02)

            lngStatus = objHoleProf.End(ValidationCriteria:=SolidEdgeConstants.ProfileValidationType.igProfileClosed)

            If lngStatus <> 0 Then
                MsgBox("Profile not closed")
            End If

            objHole = objDoc.Models.Item(1).Holes.AddThroughAll(Profile:=objHoleProf,
                    ProfilePlaneSide:=SolidEdgePart.FeaturePropertyConstants.igLeft,
                                                                Data:=objHoleData)

            objHoleProf.Visible = False

            If objHole.Status <> SolidEdgePart.FeatureStatusConstants.igFeatureOK Then
                MsgBox("Add failed")
            End If
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
using System.IO;

public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objApp = null;
        SolidEdgePart.PartDocument objDoc = null;
        SolidEdgePart.Model objModel = null;
        SolidEdgePart.HoleGeometry objHoleGeom = null;
        SolidEdgePart.HoleGeometries objHoleGeoms = null;
        SolidEdgePart.HoleDataCollection objHoleDataColl = null;
        SolidEdgePart.HoleData objHoleData = null;
        SolidEdgePart.HoleData objTmpHoleData = null;
        SolidEdgePart.Profile objHoleProf = null;
        SolidEdgePart.RefPlane objRefPln = null;
        SolidEdgePart.Hole objHole = null;
        long lngStatus = 0;
        long ii = 0;

        try
        {
            objApp = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            //sample assumes a file exists containing a Part Copy of a file that has at least one Hole feature.
            objDoc = objApp.ActiveDocument;

            objHoleDataColl = objDoc.HoleDataCollection;

            objModel = objDoc.Models.Item(1);

            objHoleGeoms = objModel.HoleGeometries;

            objHoleGeom = objHoleGeoms.Item(1);

            objHoleData = objHoleDataColl.Copy(objHoleGeom.HoleData);

            objRefPln = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:objDoc.RefPlanes.Item(3), Distance:0.0D, NormalSide:SolidEdgePart.FeaturePropertyConstants.igRight);

            objHoleProf = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:objRefPln);

            objHoleProf.Holes2d.Add(XCenter:0.04, YCenter:0.02);

            lngStatus = objHoleProf.End(ValidationCriteria:SolidEdgeConstants.ProfileValidationType.igProfileClosed);

            if (lngStatus != 0)
            {
                MessageBox.Show("Profile not closed");
            }

            objHole = objDoc.Models.Item(1).Holes.AddThroughAll(Profile:objHoleProf, ProfilePlaneSide:SolidEdgePart.FeaturePropertyConstants.igLeft, Data:objHoleData);

            objHoleProf.Visible = false;

            if (objHole.Status != SolidEdgePart.FeatureStatusConstants.igFeatureOK)
            {
                MessageBox.Show("Add failed");
            }
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }
}
