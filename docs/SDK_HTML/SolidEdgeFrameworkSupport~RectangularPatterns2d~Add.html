using System;
using System.Collections.Generic;
using System.Runtime.InteropServices;

namespace Examples
{
    class Program
    {
        [STAThread]
        static void Main(string[] args)
        {
            SolidEdgeFramework.Application application = null;
            SolidEdgeFramework.Documents documents = null;
            SolidEdgePart.PartDocument partDocument = null;
            SolidEdgePart.RefPlanes refPlanes = null;
            SolidEdgePart.ProfileSets profileSets = null;
            SolidEdgePart.ProfileSet profileSet = null;
            SolidEdgePart.Profiles profiles = null;
            SolidEdgePart.Profile profile = null;
            SolidEdgeFrameworkSupport.RectangularPatterns2d rectangularPatterns2d = null;
            SolidEdgeFrameworkSupport.RectangularPattern2d rectangularPattern2d = null;
            try
            {
                // See "Handling 'Application is Busy' and 'Call was Rejected By Callee' errors" topic.
                OleMessageFilter.Register();

                // Attempt to connect to a running instance of Solid Edge.
                application = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
                documents = application.Documents;
                partDocument = (SolidEdgePart.PartDocument)documents.Add("SolidEdge.PartDocument");
                refPlanes = partDocument.RefPlanes;
                profileSets = partDocument.ProfileSets;
                profileSet = profileSets.Add();
                profiles = profileSet.Profiles;
                profile = profiles.Add(refPlanes.Item(1));
                rectangularPatterns2d = profile.RectangularPatterns2d;

                var OffsetType = SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternFillOffset;
                rectangularPattern2d = rectangularPatterns2d.Add(0, 0, 0.1, 0.05, 0, OffsetType, 6, 4, 0.015, 0.01);

                for (int i = 1; i <= rectangularPatterns2d.Count; i++)
                {
                    rectangularPattern2d = rectangularPatterns2d.Item(i);
                }
            }
            catch (System.Exception ex)
            {
                Console.WriteLine(ex);
            }
            finally
            {
                OleMessageFilter.Unregister();
            }
        }
    }
}
