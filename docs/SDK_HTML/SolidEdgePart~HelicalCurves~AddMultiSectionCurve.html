    Sub Main()
        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objConstructions As SolidEdgePart.Constructions = Nothing

        Dim objEdges As SolidEdgeGeometry.Edges = Nothing
        Dim objEdgesCyl As SolidEdgeGeometry.Edges = Nothing
        Dim objFacesCyl As SolidEdgeGeometry.Faces = Nothing
        Dim objFaces(1) As Object
        Dim objEdge As SolidEdgeGeometry.Edge = Nothing

        Dim objEdgArr(0 To 2) As Object
        Dim kptypeArr(0 To 2) As SolidEdgeConstants.KeyPointType
        Dim PositionArray(11) As Double
        Dim PositionArrayVertex(0 To 2) As Double
        Dim pointtypeArr(0 To 2) As SolidEdgeConstants.PointTypeConstants

        Dim objHelicalCurve As SolidEdgePart.HelicalCurve = Nothing
        Dim objVertex As SolidEdgeGeometry.Vertex = Nothing
        Dim objPlane As SolidEdgePart.RefPlane = Nothing

        Dim DiameterArray(0 To 3) As Double
        Dim HeightArray(0 To 3) As Double
        Dim TurnsArray(0 To 3) As Double
        Dim PitchArray(0 To 3) As Double

        Dim MethodType As SolidEdgePart.HelicalCurveMethodType
        Dim TaperType As SolidEdgePart.HelicalCurveTaperByType
        Dim IsHelixRotationRightHanded As Boolean
        Dim IsHelixDirectionStartToEnd As Boolean
        Dim HeightArray1(4) As Double
        Dim PitchArray1(4) As Double
        Dim TurnsArray1(4) As Double
        Dim DiameterArray1(4) As Double
        Dim objProfArr(0 To 1) As SolidEdgePart.Profile
        Dim objLines As SolidEdgeFrameworkSupport.Lines2d = Nothing
        Dim objRelns1 As SolidEdgeFrameworkSupport.Relations2d = Nothing
        Dim lngStatus As Long
        Dim objModel As SolidEdgePart.Model = Nothing
        Dim objC1 As SolidEdgeFrameworkSupport.Circle2d = Nothing
        Dim objPPlane As SolidEdgePart.RefPlane = Nothing
        Dim objRPParallel As SolidEdgePart.RefPlane = Nothing

        Try
            Try
                objApp = Marshal.GetActiveObject("SolidEdge.Application")
            Catch ex As Exception
                objApp = Activator.CreateInstance(Type.GetTypeFromProgID("SolidEdge.Application"))
                objApp.DisplayAlerts = True
                objApp.Visible = True
            End Try

            objDoc = objApp.Documents.Add("SolidEdge.PartDocument")

            ' Draw the Profile
            objProfArr(0) = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objDoc.RefPlanes(1))
            objLines = objProfArr(0).Lines2d
            objLines.AddBy2Points(x1:=0, y1:=0, x2:=0.06, y2:=0)
            objLines.AddBy2Points(x1:=0.06, y1:=0, x2:=0.06, y2:=0.06)
            objLines.AddBy2Points(x1:=0.06, y1:=0.06, x2:=0, y2:=0.06)
            objLines.AddBy2Points(x1:=0, y1:=0.06, x2:=0, y2:=0)
            ' Relate the Lines to make the Profile closed
            objRelns1 = objProfArr(0).Relations2d
            objRelns1.AddKeypoint(Object1:=objLines(0), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines(1), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            objRelns1.AddKeypoint(Object1:=objLines(1), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines(2), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            objRelns1.AddKeypoint(Object1:=objLines(2), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines(3), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            objRelns1.AddKeypoint(Object1:=objLines(3), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines(0), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            ' Check for the Profile Validity
            lngStatus = objProfArr(0).End(ValidationCriteria:=SolidEdgePart.ProfileValidationType.igProfileClosed)
            If lngStatus <> 0 Then
                MsgBox("Profile not closed")
            End If
            ' Create the Base Protrusion Object
            objModel = objDoc.Models.AddFiniteExtrudedProtrusion(NumberOfProfiles:=1, _
                                                                     ProfileArray:=objProfArr, ProfilePlaneSide:=SolidEdgePart.FeaturePropertyConstants.igRight, _
                                                                     ExtrusionDistance:=0.02)
            objProfArr(0).Visible = False

            ' adding a reference plane parallel to and at a distance to a pricipal plane
            objPPlane = objDoc.RefPlanes.Item(1)
            objRPParallel = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:=objPPlane, Distance:=0.1, _
                                                                    NormalSide:=SolidEdgePart.FeaturePropertyConstants.igRight)

            ' Create a Profile
            objProfArr(1) = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objRPParallel)

            objC1 = objProfArr(1).Circles2d.AddByCenterRadius(x:=0, y:=0, Radius:=0.01)

            ' Create the Cylindrical Protrusion Object
            objModel.ExtrudedProtrusions.AddFinite(objProfArr(1), SolidEdgePart.FeaturePropertyConstants.igRight, SolidEdgePart.FeaturePropertyConstants.igLeft, 0.02)
            objProfArr(1).Visible = False

            objConstructions = objDoc.Constructions
            If Err().Number <> 0 Or objConstructions Is Nothing Then
                Exit Sub
            End If

            'obtain the first model and get all its edges
            objEdges = objDoc.Models.Item(1).ExtrudedProtrusions.Item(1).Edges(SolidEdgeConstants.FeatureTopologyQueryTypeConstants.igQueryAll)
            If Err().Number <> 0 Or objEdges Is Nothing Then
                Exit Sub
            End If

            objEdgesCyl = objDoc.Models.Item(1).ExtrudedProtrusions.Item(2).Edges(SolidEdgeConstants.FeatureTopologyQueryTypeConstants.igQueryAll)
            If Err().Number <> 0 Or objEdgesCyl Is Nothing Then
                Exit Sub
            End If

            objFacesCyl = objDoc.Models.Item(1).ExtrudedProtrusions.Item(2).Faces(SolidEdgeConstants.FeatureTopologyQueryTypeConstants.igQueryCylinder)
            If Err().Number <> 0 Or objFacesCyl Is Nothing Then
                Exit Sub
            End If

            'fill the edges in array
            ' Get the Edges and store them in an Array
            objEdge = objEdges.Item(1)
            objVertex = objEdge.StartVertex
            objVertex.GetPointData(PositionArrayVertex)
            objEdgArr(0) = objVertex

            objEdgArr(1) = Nothing

            objEdge = objEdges(3)
            objVertex = objEdge.StartVertex
            objEdgArr(2) = objVertex

            objPlane = objDoc.RefPlanes(0)

            'Define an Array of Points
            PositionArray(0) = 0
            PositionArray(1) = 0
            PositionArray(2) = 0
            PositionArray(3) = 0.02
            PositionArray(4) = 0.02
            PositionArray(5) = 0.02
            PositionArray(6) = 0
            PositionArray(7) = 0
            PositionArray(8) = 0

            'now fill the point type aaray
            pointtypeArr(0) = SolidEdgeConstants.PointTypeConstants.igKeyPoint
            pointtypeArr(1) = SolidEdgeConstants.PointTypeConstants.igSpacePoint
            pointtypeArr(2) = SolidEdgeConstants.PointTypeConstants.igKeyPoint
   
            DiameterArray(0) = 0.2
            HeightArray(0) = 0.0
            TurnsArray(0) = 0.0
            PitchArray(0) = 0.01

            DiameterArray(1) = 0.8
            HeightArray(1) = 0.1
            TurnsArray(1) = 2.0
            PitchArray(1) = 0.05

            DiameterArray(2) = 0.2
            HeightArray(2) = 0.3
            TurnsArray(2) = 6.0
            PitchArray(2) = 0.05

            DiameterArray(3) = 1.2
            HeightArray(3) = 0.5
            TurnsArray(3) = 10.0
            PitchArray(3) = 0.05

            objHelicalCurve = objConstructions.HelicalCurves.AddMultiSectionCurve(Nothing,
                                                                       3,
                                                                       pointtypeArr,
                                                                       objEdgArr,
                                                                       PositionArray,
                                                                       Nothing,
                                                                       True,
                                                                       SolidEdgePart.HelicalCurveMethodType.igHelicalCurveMethodPitchAndTurns,
                                                                       SolidEdgePart.HelicalCurveTaperByType.igHelicalCurveTaperByDiameter,
                                                                       True,
                                                                       True,
                                                                       True,
                                                                       3,
                                                                       DiameterArray,
                                                                       HeightArray,
                                                                       TurnsArray,
                                                                       PitchArray)

            objHelicalCurve.EditMultiSectionHelicalCurve(SolidEdgePart.HelicalCurveMethodType.igHelicalCurveMethodPitchAndTurns, 2, 0.0, 0.1, 5, 0.2)
            If Err().Number <> 0 Or objHelicalCurve Is Nothing Then
                Exit Sub
            End If

            objHelicalCurve.getHelicalCurveData(3, MethodType, TaperType, IsHelixRotationRightHanded, IsHelixDirectionStartToEnd, HeightArray1, PitchArray1, TurnsArray1, DiameterArray1)
            If Err().Number <> 0 Or objHelicalCurve Is Nothing Then
                Exit Sub
            End If

        Catch ex As Exception
            Console.WriteLine(ex.Message)
        End Try




    End Sub

End Module
