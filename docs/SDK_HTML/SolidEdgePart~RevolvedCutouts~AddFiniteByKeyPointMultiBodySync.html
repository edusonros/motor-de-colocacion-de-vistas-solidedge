    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objModel1 As SolidEdgePart.Model = Nothing
        Dim objModel2 As SolidEdgePart.Model = Nothing
        Dim objModel3 As SolidEdgePart.Model = Nothing
        Dim objRevCuts As SolidEdgePart.RevolvedCutouts = Nothing
        Dim objRevCut As SolidEdgePart.RevolvedCutout = Nothing
        Dim objBody As SolidEdgeGeometry.Body = Nothing
        Dim objProfile1(1) As SolidEdgePart.Profile
        Dim objProfile2(1) As SolidEdgePart.Profile
        Dim objProfile3(1) As SolidEdgePart.Profile
        Dim objProfile4(1) As SolidEdgePart.Profile
        Dim objRefAxisProfile As SolidEdgePart.Profile = Nothing
        Dim objCircles As SolidEdgeFrameworkSupport.Circles2d = Nothing
        Dim objLines As SolidEdgeFrameworkSupport.Lines2d = Nothing
        Dim objLines1 As SolidEdgeFrameworkSupport.Lines2d = Nothing
        Dim objRelns As SolidEdgeFrameworkSupport.Relations2d = Nothing
        Dim objRelns1 As SolidEdgeFrameworkSupport.Relations2d = Nothing
        Dim objRefPln As SolidEdgePart.RefPlane

        Try
            ' Create/get the application with specific settings
            Try
                objApp = GetObject(, "SolidEdge.Application")
            Catch ex As Exception
                objApp = CreateObject("SolidEdge.Application")
                objApp.Visible = True
            Finally
                objDoc = objApp.Documents.Add("SolidEdge.PartDocument")
                objDoc.ModelingMode = SolidEdgePart.ModelingModeConstants.seModelingModeOrdered
            End Try

            ' Create an extruded protrusion in each of three different design bodies.

            Call objDoc.Models.AddBody(SolidEdgePart.AddBodyTypeConstants.igPartType, "Design Body_1")

            ' Create the profile for the first protrusion

            objProfile1(0) = objDoc.Sketches.AddByPlane(objDoc.RefPlanes.Item(3)).Profile

            objLines = objProfile1(0).Lines2d
            objRelns = objProfile1(0).Relations2d

            Call objLines.AddBy2Points(x1:=0, y1:=0, x2:=0.08, y2:=0)
            Call objLines.AddBy2Points(x1:=0.08, y1:=0, x2:=0.08, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.08, y1:=0.06, x2:=0.064, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.064, y1:=0.06, x2:=0.064, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.064, y1:=0.02, x2:=0.048, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.048, y1:=0.02, x2:=0.048, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.048, y1:=0.06, x2:=0.032, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.032, y1:=0.06, x2:=0.032, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.032, y1:=0.02, x2:=0.016, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.016, y1:=0.02, x2:=0.016, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.016, y1:=0.06, x2:=0, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0, y1:=0.06, x2:=0, y2:=0)

            Call objRelns.AddKeypoint(Object1:=objLines.Item(1), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(2), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(2), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(3), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(3), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(4), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(4), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(5), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(5), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(6), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(6), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(7), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(7), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(8), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(8), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(9), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(9), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(10), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(10), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(11), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(11), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(12), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(12), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(1), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)

            ' Check for Profile validity
            lngStatus = objProfile1(0).End(ValidationCriteria:=SolidEdgePart.ProfileValidationType.igProfileClosed)

            If lngStatus <> 0 Then
                MsgBox(("Profile not closed"))
            Else
                objProfile1(0).Visible = False
            End If

            ' Create the Extruded Protrusion Feature

            objModel1 = objDoc.Models.AddFiniteExtrudedProtrusion(NumberOfProfiles:=1,
                                                          ProfileArray:=objProfile1,
                                                          ProfilePlaneSide:=SolidEdgePart.FeaturePropertyConstants.igRight,
                                                          ExtrusionDistance:=0.1)

            If objModel1.ExtrudedProtrusions.Item(1) Is Nothing Then
                MsgBox(("Error in the Creation of Base Protrusion Feature object"))
            End If

            ' Create a new design body and the same protrusion, but with the profile translated by 100 mm in the global positive X direction

            Call objDoc.Models.AddBody(SolidEdgePart.AddBodyTypeConstants.igPartType, "Design Body_2")

            objProfile2(0) = objDoc.Sketches.AddByPlane(objDoc.RefPlanes.Item(3)).Profile

            objLines = objProfile2(0).Lines2d
            objRelns = objProfile2(0).Relations2d

            Call objLines.AddBy2Points(x1:=0.1, y1:=0, x2:=0.18, y2:=0)
            Call objLines.AddBy2Points(x1:=0.18, y1:=0, x2:=0.18, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.18, y1:=0.06, x2:=0.164, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.164, y1:=0.06, x2:=0.164, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.164, y1:=0.02, x2:=0.148, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.148, y1:=0.02, x2:=0.148, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.148, y1:=0.06, x2:=0.132, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.132, y1:=0.06, x2:=0.132, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.132, y1:=0.02, x2:=0.116, y2:=0.02)
            Call objLines.AddBy2Points(x1:=0.116, y1:=0.02, x2:=0.116, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.116, y1:=0.06, x2:=0.1, y2:=0.06)
            Call objLines.AddBy2Points(x1:=0.1, y1:=0.06, x2:=0.1, y2:=0)

            Call objRelns.AddKeypoint(Object1:=objLines.Item(1), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(2), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(2), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(3), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(3), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(4), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(4), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(5), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(5), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(6), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(6), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(7), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(7), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(8), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(8), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(9), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(9), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(10), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(10), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(11), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(11), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(12), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            Call objRelns.AddKeypoint(Object1:=objLines.Item(12), Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objLines.Item(1), Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)

            lngStatus = objProfile2(0).End(ValidationCriteria:=SolidEdgePart.ProfileValidationType.igProfileClosed)

            If lngStatus <> 0 Then
                MsgBox(("Profile not closed"))
            Else
                objProfile2(0).Visible = False
            End If


            ' Create the second Extruded Protrusion Feature

            objModel2 = objDoc.Models.AddFiniteExtrudedProtrusion(NumberOfProfiles:=1,
                                                          ProfileArray:=objProfile2,
                                                          ProfilePlaneSide:=SolidEdgePart.FeaturePropertyConstants.igRight,
                                                          ExtrusionDistance:=0.2)

            If objModel2.ExtrudedProtrusions.Item(1) Is Nothing Then
                MsgBox(("Error in the Creation of Base Protrusion Feature object"))
            End If

            ' Now add the third design body and create the same protrusion, but offset 200 mm along X from the global origin.

            Call objDoc.Models.AddBody(SolidEdgePart.AddBodyTypeConstants.igPartType, "Design Body_3")

            objProfile3(0) = objDoc.Sketches.AddByPlane(objDoc.RefPlanes.Item(3)).Profile

            objLines = objProfile3(0).Lines2d
            objRelns = objProfile3(0).Relations2d

            objRevCut = objRevCuts.AddFiniteByKeyPointMultiBodySync(NumberOfProfiles:nProfiles, ProfileArray:objProfile4, RefAxis:objRefAxis, KeyPointOrTangentFace:objKeyPointVertex, KeyPointFlags:SolidEdgePart.KeyPointExtentConstants.igTangentNormal, ProfileSide:SolidEdgePart.FeaturePropertyConstants.igLeft, ProfilePlaneSide:SolidEdgePart.FeaturePropertyConstants.igRight, NumberOfBodies:nBodies, BodyArray:Bodies);

            if (objRevCut == null)
            {
                MessageBox.Show("AddFiniteByKeyPointMultiBodyMethod with ProfileSide set to igLeft and ProfilePlaneSide set to igRight failed");
            }

            if (objRevCut.Status != SolidEdgePart.FeatureStatusConstants.igFeatureOK)
            {
                MessageBox.Show("AddFiniteByKeyPointMultiBodyMethod created a failed feature for ProfileSide=igLeft and ProfilePlaneSide=igRight");
            }


        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }

    }
