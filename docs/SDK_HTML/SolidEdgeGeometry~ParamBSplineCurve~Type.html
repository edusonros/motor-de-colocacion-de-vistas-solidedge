using System;
using System.Runtime.InteropServices;

namespace Examples
{
    class Program
    {
        [STAThread]
        static void Main(string[] args)
        {
            SolidEdgeFramework.Application application = null;
            SolidEdgePart.PartDocument partDocument = null;
            SolidEdgePart.Models models = null;
            SolidEdgePart.Model model = null;
            SolidEdgeGeometry.Body body = null;
            SolidEdgeGeometry.Shells shells = null;
            SolidEdgeGeometry.Shell shell = null;
            SolidEdgeGeometry.Faces faces = null;
            SolidEdgeGeometry.Face face = null;
            SolidEdgeGeometry.Loops loops = null;
            SolidEdgeGeometry.Loop loop = null;
            SolidEdgeGeometry.EdgeUses edgeUses = null;
            SolidEdgeGeometry.EdgeUse edgeUse = null;
            SolidEdgeGeometry.ParamBSplineCurve paramBSplineCurve = null;

            try
            {
                // See "Handling 'Application is Busy' and 'Call was Rejected By Callee' errors" topic.
                OleMessageFilter.Register();

                // Attempt to connect to a running instance of Solid Edge.
                application = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
                partDocument = application.ActiveDocument as SolidEdgePart.PartDocument;

                if (partDocument != null)
                {
                    models = partDocument.Models;
                    model = models.Item(1);
                    body = (SolidEdgeGeometry.Body)model.Body;
                    shells = (SolidEdgeGeometry.Shells)body.Shells;
                    shell = (SolidEdgeGeometry.Shell)shells.Item(1);
                    faces = (SolidEdgeGeometry.Faces)shell.Faces;
                    face = (SolidEdgeGeometry.Face)faces.Item(1);
                    loops = (SolidEdgeGeometry.Loops)face.Loops;
                    loop = (SolidEdgeGeometry.Loop)loops.Item(1);
                    edgeUses = (SolidEdgeGeometry.EdgeUses)loop.EdgeUses;
                    edgeUse = (SolidEdgeGeometry.EdgeUse)edgeUses.Item(1);
                    paramBSplineCurve = edgeUse.Geometry as SolidEdgeGeometry.ParamBSplineCurve;

                    if (paramBSplineCurve != null)
                    {
                        var paramBSplineCurveType = paramBSplineCurve.Type;
                    }
                }
            }
            catch (System.Exception ex)
            {
                Console.WriteLine(ex);
            }
            finally
            {
                OleMessageFilter.Unregister();
            }
        }
    }
}
