Imports System.Runtime.InteropServices

Public Class Form1

    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApplication As SolidEdgeFramework.Application = Nothing
        Dim objPartDoc As SolidEdgePart.PartDocument = Nothing
        Dim objEdges(0 To 4) As Object 
        Dim objEdges2(4) As Object 
        Dim objVertex As Object = Nothing 
        Dim objVertex2 As SolidEdgeGeometry.Vertex = Nothing
        Dim objFeatArray(0 To 4) As Object 
        Dim objPattern As SolidEdgePart.Pattern = Nothing
        Dim objPlane As SolidEdgeGeometry.Plane = Nothing
        Dim SkipCount As Integer
        Dim objProfile As SolidEdgePart.Profile = Nothing
        Dim objTrainingFolder As DirectoryInfo
        Dim objSketch As SolidEdgePart.Sketch = Nothing
        Try
            objSEApplication = Marshal.GetActiveObject("SolidEdge.Application")

            'Get Training Folder
            objTrainingFolder = GetTrainingFolder()
            'Open C:\Program Files\Solid Edge STXX\Training\AllenScrewM8.par
            objSEApplication.Documents.Open(Path.Combine(objTrainingFolder.FullName, "AllenScrewM8.par"))
            objPartDoc = objSEApplication.ActiveDocument
            objSketch = objPartDoc.Sketches.Add()
            objProfile = objSketch.Profiles.Add(objPartDoc.RefPlanes.Item(1))
            objProfile.Lines2d.AddBy2Points(0.0, 0.0, 0.5, 0.0)
            ' Switch to ISO view.
            objSEApplication.StartCommand(SolidEdgeConstants.PartCommandConstants.PartViewISOView)
            objEdges(0) = objProfile.CurveBody.Curves
            objVertex = objProfile.CurveBody.CurveVertices.Item(1)
            objFeatArray(0) = objPartDoc.Models.Item(1).ExtrudedProtrusions.Item(1)

            ' Add pattern with skip count 1
            objPattern = objPartDoc.Models.Item(1).Patterns.AddByCurveEx(1, objFeatArray, _
                                                                     SolidEdgePart.PatternTypeConstants.seSmartPattern, 1, _
                                                                     objEdges, objVertex, SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveRightSide, _
                                                                     0, SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternChordLengthOffset, _
                                                                     4, 0.1, 0, objEdges2, objVertex2, SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveRightSide, _
                                                                     0, SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternChordLengthOffset, 0, 0, _
                                                                     SolidEdgePart.PatternTransformTypeConstants.sePatternTransformFullRotation, _
                                                                     SolidEdgePart.PatternTransformRotateTypeConstants.sePatternTransformRotateOnCurvePosition, objPlane, 1)

            'Get skip count
            SkipCount = objPattern.SkipCount()
            SkipCount = 2
            objPattern.SkipCount = 2
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
    Function GetTrainingFolder() As DirectoryInfo
        Dim objInstallData As SEInstallDataLib.SEInstallData = Nothing
        Dim objInstallFolder As DirectoryInfo = Nothing
        Dim objTrainingFolder As DirectoryInfo = Nothing

        Try
            objInstallData = New SEInstallDataLib.SEInstallData
            objInstallFolder = New DirectoryInfo(objInstallData.GetInstalledPath())
            objTrainingFolder = New DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"))
        Catch
        Finally
            If Not (objInstallData Is Nothing) Then
                Marshal.FinalReleaseComObject(objInstallData)
                objInstallData = Nothing
            End If
        End Try

        Return objTrainingFolder
    End Function

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;
using System.IO;
public class Form1
{

    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objSEApplication = null;
        SolidEdgePart.PartDocument objPartDoc = null;
        object[] objEdges = new object[5];
        object[] objEdges2 = new object[5];
        object objVertex = null;
        SolidEdgeGeometry.Vertex objVertex2 = null;
        object[] objFeatArray = new object[5];
        SolidEdgePart.Pattern objPattern = null;
        SolidEdgeGeometry.Plane objPlane = null;
        int SkipCount = 0;
        SolidEdgePart.Profile objProfile = null;
        DirectoryInfo objTrainingFolder = null;
        SolidEdgePart.Sketch objSketch = null;
        try
        {
            objSEApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");

            //Get Training Folder
            objTrainingFolder = GetTrainingFolder();
            //Open C:\Program Files\Solid Edge STXX\Training\AllenScrewM8.par
            objSEApplication.Documents.Open(Path.Combine(objTrainingFolder.FullName, "AllenScrewM8.par"));
            objPartDoc = objSEApplication.ActiveDocument;
            objSketch = objPartDoc.Sketches.Add();
            objProfile = objSketch.Profiles.Add(objPartDoc.RefPlanes.Item(1));
            objProfile.Lines2d.AddBy2Points(0.0, 0.0, 0.5, 0.0);
            // Switch to ISO view.
            objSEApplication.StartCommand(SolidEdgeConstants.PartCommandConstants.PartViewISOView);
            objEdges[0] = objProfile.CurveBody.Curves;
            objVertex = objProfile.CurveBody.CurveVertices.Item(1);
            objFeatArray[0] = objPartDoc.Models.Item(1).ExtrudedProtrusions.Item(1);

            // Add pattern with skip count 1
            objPattern = objPartDoc.Models.Item(1).Patterns.AddByCurveEx(1, objFeatArray, SolidEdgePart.PatternTypeConstants.seSmartPattern, 1, objEdges, objVertex, SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveRightSide, 0, SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternChordLengthOffset, 4, 0.1, 0, objEdges2, objVertex2, SolidEdgePart.PatternCurveAnchorSideConstants.sePatternCurveRightSide, 0, SolidEdgeFrameworkSupport.PatternOffsetTypeConstants.sePatternChordLengthOffset, 0, 0, SolidEdgePart.PatternTransformTypeConstants.sePatternTransformFullRotation, SolidEdgePart.PatternTransformRotateTypeConstants.sePatternTransformRotateOnCurvePosition, objPlane, 1);

            //Get skip count
            SkipCount = objPattern.SkipCount();
            SkipCount = 2;
            objPattern.SkipCount = 2;
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }
    public DirectoryInfo GetTrainingFolder()
    {
        SEInstallDataLib.SEInstallData objInstallData = null;
        DirectoryInfo objInstallFolder = null;
        DirectoryInfo objTrainingFolder = null;

        try
        {
            objInstallData = new SEInstallDataLib.SEInstallData();
            objInstallFolder = new DirectoryInfo(objInstallData.GetInstalledPath());
            objTrainingFolder = new DirectoryInfo(Path.Combine(objInstallFolder.Parent.FullName, "Training"));
        }
        catch
        {
        }
        finally
        {
            if (objInstallData != null)
            {
                Marshal.FinalReleaseComObject(objInstallData);
                objInstallData = null;
            }
        }

        return objTrainingFolder;
    }
