Specifies the number of Section objects used in the construction of the lofted protrusion feature.
The size of the array is equal to the value of NumSections.
The size of the array is equal to the value of NumSections. The two cross-section types are igProfileBasedCrossSection and igEdgeBasedCrossSection constants.
The size of the array is equal to the value of NumSections. For models that are created with periodic cross-sections that include objects such as circles, ellipses, and closed B-Splines, Solid Edge automatically selects the traverse points, so the value of Origins may be set to 0,0 or the cross-sections can be skipped without the origins defined. If the origins are not defined for a cross-section, make sure the cross-section is counted in the array, so that the other cross-sections origins are represented correctly to in the array.

Remarks
Typically, this method is used to create a protrusion that is to be the base feature for a model. Additional lofted protrusions added to the model are created through methods that belong to the model's LoftedProtrusions collection object. This method fails if called more than one time because only one base feature (or Model) is valid in a Part file.
Example
  • Example (Visual Basic) 
Private Sub Form_Load()
    Dim objApp As SolidEdgeFramework.Application
    Dim objDoc As SolidEdgePart.PartDocument
    Dim objRP1 As SolidEdgePart.RefPlane
    Dim objProf1 As SolidEdgePart.Profile
    Dim objProf2 As SolidEdgePart.Profile
    Dim obj2Lines As SolidEdgeFrameworkSupport.Lines2d
    Dim obj2Relns As SolidEdgeFrameworkSupport.Relations2d
    Dim objRP3 As SolidEdgePart.RefPlane
    Dim objProf3 As SolidEdgePart.Profile
    Dim objCSArray(1 To 3) As SolidEdgePart.Profile
    Dim objModel As SolidEdgePart.Model
    Dim lngStatus As Long
    Dim lngCSType(1 To 3) As Long
    Dim varOrigins(1 To 3) As Variant
    Dim dblOrigin(1 To 2) As Double
    ' Report errors
    Const PI = 3.14159265358979
    ' Create/get the application with specific settings
    On Error Resume Next
    Set objApp = GetObject(, "SolidEdge.Application")
    If Err Then
        Err.Clear
        Set objApp = CreateObject("SolidEdge.Application")
        Set objDoc = objApp.Documents.Add("SolidEdge.PartDocument")
        objApp.Visible = True
    Else
        Set objDoc = objApp.ActiveDocument
    End If
    On Error GoTo 0
    ' *** creating a base lofted protrusion feature
    ' creating the profiles and validating them
    Set objRP1 = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:=objDoc.RefPlanes(2), _
                                                        Distance:=0.05, NormalSide:=igLeft)
    Set objProf1 = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objRP1)
    Call objProf1.Ellipses2d.AddByCenter(xCenter:=0, yCenter:=0, _
                                         xMajor:=0.04, yMajor:=0, Ratio:=0.5, Orientation:=igGeom2dOrientClockwise)
    lngStatus = objProf1.End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox "Elliptical profile for the base feature is not closed"
    End If
    objProf1.Visible = False
    Set objProf2 = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objDoc.RefPlanes(2))
    Set obj2Lines = objProf2.Lines2d
    Call obj2Lines.AddBy2Points(x1:=0.025, y1:=0.025, x2:=-0.025, y2:=0.025)
    Call obj2Lines.AddBy2Points(x1:=-0.025, y1:=0.025, x2:=-0.025, y2:=-0.025)
    Call obj2Lines.AddBy2Points(x1:=-0.025, y1:=-0.025, x2:=0.025, y2:=-0.025)
    Call obj2Lines.AddBy2Points(x1:=0.025, y1:=-0.025, x2:=0.025, y2:=0.025)
    Set obj2Relns = objProf2.Relations2d
    Call obj2Relns.AddKeypoint(object1:=obj2Lines(1), index1:=igLineEnd, object2:=obj2Lines(2), index2:=igLineStart)
    Call obj2Relns.AddKeypoint(object1:=obj2Lines(2), index1:=igLineEnd, object2:=obj2Lines(3), index2:=igLineStart)
    Call obj2Relns.AddKeypoint(object1:=obj2Lines(3), index1:=igLineEnd, object2:=obj2Lines(4), index2:=igLineStart)
    Call obj2Relns.AddKeypoint(object1:=obj2Lines(4), index1:=igLineEnd, object2:=obj2Lines(1), index2:=igLineStart)
    lngStatus = objProf2.End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox "Rectangular profile for the base feature is not closed"
    End If
    Call objProf2.Convert3DCoordinate(x3d:=0, y3d:=0.025, z3d:=0.025, _
                                      x2d:=dblOrigin(1), y2d:=dblOrigin(2))
    objProf2.Visible = False
    Set objRP3 = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:=objDoc.RefPlanes(2), _
                                                        Distance:=0.05, NormalSide:=igRight)
    Set objProf3 = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objRP3)
    Call objProf3.Circles2d.AddByCenterRadius(x:=0, y:=0, Radius:=0.025)
    lngStatus = objProf3.End(ValidationCriteria:=igProfileClosed)
    If lngStatus <> 0 Then
        MsgBox "Circular profile for the base feature is not closed"
    End If
    objProf3.Visible = False
    ' *** creating the loft protrusion feature
    ' setting the constants
    Set objCSArray(1) = objProf1
    Set objCSArray(2) = objProf2
    Set objCSArray(3) = objProf3
    lngCSType(1) = igProfileBasedCrossSection
    lngCSType(2) = igProfileBasedCrossSection
    lngCSType(3) = igProfileBasedCrossSection
    varOrigins(1) = 0
    varOrigins(2) = dblOrigin
    varOrigins(3) = 0
    Set objModel = objDoc.Models.AddLoftedProtrusion(NumSections:=3, _
                                                     CrossSections:=objCSArray, CrossSectionTypes:=lngCSType, _
                                                     origins:=varOrigins, SegmentMaps:=0, MaterialSide:=igLeft, _
                                                     StartExtentType:=igNone, StartExtentDistance:=0, _
                                                     StartSurfaceOrRefPlane:=Nothing, EndExtentType:=igNone, _
                                                     EndExtentDistance:=0, EndSurfaceOrRefPlane:=Nothing, _
                                                     StartTangentType:=igNone, StartTangentMagnitude:=0, _
                                                     EndTangentType:=igNone, EndTangentMagnitude:=0)



    ' USER DISPLAY
    ' Release objects
    Set objApp = Nothing
    Set objDoc = Nothing
    Set objRP1 = Nothing
    Set objProf1 = Nothing
    Set objProf2 = Nothing
    Set obj2Lines = Nothing
    Set obj2Relns = Nothing
    Set objRP3 = Nothing
    Set objProf3 = Nothing
    Set objCSArray(1) = Nothing
    Set objCSArray(2) = Nothing
    Set objCSArray(3) = Nothing
    Set objModel = Nothing
End Sub
