Imports System.Runtime.InteropServices

Public Class Form1
    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim application As SolidEdgeFramework.Application = Nothing
        Dim partDocument As SolidEdgePart.PartDocument = Nothing
        Dim draftDocument As SolidEdgeDraft.DraftDocument = Nothing
        Dim objLine1 As SolidEdgeFrameworkSupport.Line2d = Nothing
        Dim objLine2 As SolidEdgeFrameworkSupport.Line2d = Nothing

        Dim objLine3 As SolidEdgeFrameworkSupport.Line2d = Nothing
        Dim objLine4 As SolidEdgeFrameworkSupport.Line2d = Nothing

        Dim objDims As SolidEdgeFrameworkSupport.Dimensions = Nothing
        Dim objDim As SolidEdgeFrameworkSupport.Dimension = Nothing

        Try
            application = Marshal.GetActiveObject("SolidEdge.Application")
            draftDocument = application.Documents.Add("SolidEdge.DraftDocument")
            objDims = draftDocument.ActiveSheet.Dimensions
            objLine1 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.1, 0.1, 0.3, 0.3)
            objLine2 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.2, 0.1, 0.4, 0.5)

            objLine3 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.5, 0.1, 0.7, 0.3)
            objLine4 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.6, 0.1, 0.8, 0.5)

            objDim = objDims.AddDistanceIntersectionToIntersection(objLine1, 0.1, 0.1, 0.0, objLine2, 0.2, 0.1, 0.0, objLine3, 0.5, 0.5, 0.0, objLine4, 0.6, 0.5, 0.0)
        Catch ex As Exception
            MsgBox(ex.ToString)
        End Try
    End Sub
End Class

using System;
using System.Windows.Forms;
using System.Runtime.InteropServices;

public class Form1
{
    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application application = null;
        SolidEdgePart.PartDocument partDocument = null;
        SolidEdgeDraft.DraftDocument draftDocument = null;
        SolidEdgeFrameworkSupport.Line2d objLine1 = null;
        SolidEdgeFrameworkSupport.Line2d objLine2 = null;

        SolidEdgeFrameworkSupport.Line2d objLine3 = null;
        SolidEdgeFrameworkSupport.Line2d objLine4 = null;

        SolidEdgeFrameworkSupport.Dimensions objDims = null;
        SolidEdgeFrameworkSupport.Dimension objDim = null;

        try
        {
            application = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            draftDocument = application.Documents.Add("SolidEdge.DraftDocument");
            objDims = draftDocument.ActiveSheet.Dimensions;
            objLine1 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.1, 0.1, 0.3, 0.3);
            objLine2 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.2, 0.1, 0.4, 0.5);

            objLine3 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.5, 0.1, 0.7, 0.3);
            objLine4 = draftDocument.ActiveSheet.Lines2d.AddBy2Points(0.6, 0.1, 0.8, 0.5);

            objDim = objDims.AddDistanceIntersectionToIntersection(objLine1, 0.1, 0.1, 0.0, objLine2, 0.2, 0.1, 0.0, objLine3, 0.5, 0.5, 0.0, objLine4, 0.6, 0.5, 0.0);
        }
        catch (Exception ex)
        {
            MessageBox.Show(ex.ToString());
        }
    }
}
