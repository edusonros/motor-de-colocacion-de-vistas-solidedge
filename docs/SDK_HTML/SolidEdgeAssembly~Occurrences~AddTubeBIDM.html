Description
Creates a new Occurrence object in the Assembly document. This occurrence behaves like any other occurrence in the Assembly file, except it successfully returns a Tube object when the Tube property is invoked on the Occurrence.

Public Class Form1
    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim objSEApplication As SolidEdgeFramework.Application = Nothing
        Dim objAsmdoc As SolidEdgeAssembly.AssemblyDocument = Nothing
        Dim occurrences As SolidEdgeAssembly.Occurrences = Nothing
        Dim tube As SolidEdgeAssembly.Occurrence = Nothing
        Dim lineSegments As SolidEdgeAssembly.LineSegments = Nothing
        Dim lineSegment As SolidEdgeAssembly.LineSegment = Nothing
        Dim result As Integer = Nothing
        Dim prefix As String = Nothing
        Dim docNums As Object() = Nothing
        Dim newFileName As String = Nothing
        Dim RevisionID As String = Nothing
        Dim objectSegments(0) As Object
        Dim Title As String = Nothing
        Dim m_sVaultPath As String = "C:\Temp\New"
        Dim StartPt(0 To 2) As Double
        Dim EndPt(0 To 2) As Double
        Dim iVal As Boolean = True

        Try
            objSEApplication = CType(GetObject(, "SolidEdge.Application"), SolidEdgeFramework.Application)
            objSEApplication.Get_Set_UseBiDM_SEOption("False", iVal)
            objSEApplication.Get_Set_FileNamingRule("False", iVal)
            result = objSEApplication.GetNextDocumentNumbers(2, prefix, docNums)
            If result <> 0 Then
                Throw New Exception
            End If

            objAsmdoc = objSEApplication.Documents.Add("SolidEdge.AssemblyDocument")

            occurrences = objAsmdoc.Occurrences
            lineSegments = objAsmdoc.LineSegments
            StartPt(0) = 0.0
            StartPt(1) = 0.0
            StartPt(2) = 0.0
            EndPt(0) = 0.0
            EndPt(1) = 0.0
            EndPt(2) = 0.1
            lineSegment = lineSegments.Add(StartPt, EndPt)

            objectSegments(0) = lineSegment

            objAsmdoc.SaveAsBIDM(m_sVaultPath, docNums(0), RevisionID, Title)
            tube = occurrences.AddTubeBIDM(objectSegments, m_sVaultPath, docNums(1), RevisionID, Title, newFileName)
            objAsmdoc.Save()

        Catch ex As Exception
            occurrences = Nothing
            lineSegments = Nothing
            lineSegment = Nothing
            objectSegments(0) = Nothing
            objectSegments = Nothing
            Array.Clear(docNums, 0, docNums.Count)
            docNums = Nothing

        End Try

    End Sub

using System;

public class Form1
{
    private void Button1_Click(object sender, EventArgs e)
    {
        SolidEdgeFramework.Application objSEApplication = null;
        SolidEdgeAssembly.AssemblyDocument objAsmdoc = null;
        SolidEdgeAssembly.Occurrences occurrences = null;
        SolidEdgeAssembly.Occurrence tube = null;
        SolidEdgeAssembly.LineSegments lineSegments = null;
        SolidEdgeAssembly.LineSegment lineSegment = null;
        int result = 0;
        string prefix = null;
        object[] docNums = null;
        string newFileName = null;
        string RevisionID = null;
        object[] objectSegments = new object[1];
        string Title = null;
        string m_sVaultPath = "C:\\Temp\\New";
        double[] StartPt = new double[3];
        double[] EndPt = new double[3];
        bool iVal = true;

        try
        {
            objSEApplication = (SolidEdgeFramework.Application)Microsoft.VisualBasic.Interaction.GetObject(null, "SolidEdge.Application");
            objSEApplication.Get_Set_UseBiDM_SEOption("False", iVal);
            objSEApplication.Get_Set_FileNamingRule("False", iVal);
            result = objSEApplication.GetNextDocumentNumbers(2, prefix, docNums);
            if (result != 0)
            {
                throw new Exception();
            }

            objAsmdoc = objSEApplication.Documents.Add("SolidEdge.AssemblyDocument");

            occurrences = objAsmdoc.Occurrences;
            lineSegments = objAsmdoc.LineSegments;
            StartPt[0] = 0.0;
            StartPt[1] = 0.0;
            StartPt[2] = 0.0;
            EndPt[0] = 0.0;
            EndPt[1] = 0.0;
            EndPt[2] = 0.1;
            lineSegment = lineSegments.Add(StartPt, EndPt);

            objectSegments[0] = lineSegment;

            objAsmdoc.SaveAsBIDM(m_sVaultPath, docNums[0], RevisionID, Title);
            tube = occurrences.AddTubeBIDM(objectSegments, m_sVaultPath, docNums[1], RevisionID, Title, newFileName);
            objAsmdoc.Save();

        }
        catch (Exception ex)
        {
            occurrences = null;
            lineSegments = null;
            lineSegment = null;
            objectSegments[0] = null;
            objectSegments = null;
            Array.Clear(docNums, 0, docNums.Count());
            docNums = null;

        }

    }

    public Form1()
    {
        SubscribeToEvents();
    }


    private bool EventsSubscribed = false;
    private void SubscribeToEvents()
    {
        if (EventsSubscribed)
            return;
        else
            EventsSubscribed = true;

        Button1.Click += Button1_Click;
    }

}
