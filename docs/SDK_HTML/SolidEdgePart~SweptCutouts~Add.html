Imports System.Runtime.InteropServices

Module Module1

    Sub Main()
        Dim objSEApp As SolidEdgeFramework.Application = Nothing
        Dim objDoc As SolidEdgePart.PartDocument = Nothing
        Dim objBaseProfile(0 To 2) As SolidEdgePart.Profile
        Dim objLine1 As SolidEdgeFrameworkSupport.Line2d = Nothing
        Dim objLine2 As SolidEdgeFrameworkSupport.Line2d = Nothing
        Dim objArc1 As SolidEdgeFrameworkSupport.Arc2d = Nothing
        Dim objArc2 As SolidEdgeFrameworkSupport.Arc2d = Nothing
        Dim objModel As SolidEdgePart.Model
        Dim objExtProtrusion As SolidEdgePart.ExtrudedProtrusion = Nothing
        Dim objRelations As SolidEdgeFrameworkSupport.Relations2d = Nothing
        Dim objTraceRefPlane As SolidEdgePart.RefPlane = Nothing
        Dim objTraceprofile(0 To 2) As SolidEdgePart.Profile
        Dim objCSRefPlane As SolidEdgePart.RefPlane = Nothing
        Dim objCSProfile(0 To 2) As SolidEdgePart.Profile
        Dim vOriginArray(2) As Object
        Dim objSweptCutOut As SolidEdgePart.SweptCutout = Nothing
        Dim lngStatus As Long
        Dim lngTraceCurveTypes(0 To 2) As Long
        Dim lngCrossSectionTypes(0 To 2) As Long

        Try
            Console.WriteLine("Starting Solid Edge")
            ' Create/get the application with specific settings
            objSEApp = Marshal.GetActiveObject("SolidEdge.Application") ' Activator.CreateInstance(Type.GetTypeFromProgID("SolidEdge.Application"))
            Console.WriteLine("Creating new part file")
            objDoc = objSEApp.Documents.Add("SolidEdge.PartDocument")
            objSEApp.Visible = True
            'Create an Extruded Protrusion as Base feature
            objBaseProfile(0) = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objDoc.RefPlanes(1))
            objLine1 = objBaseProfile(0).Lines2d.AddBy2Points(x1:=0.05, y1:=0, x2:=0.05, y2:=0.15)
            objArc1 = objBaseProfile(0).Arcs2d.AddByCenterStartEnd(xCenter:=0.025, yCenter:=0.15, _
                                                                       xStart:=0.05, yStart:=0.15, xEnd:=0, yEnd:=0.15)
            objLine2 = objBaseProfile(0).Lines2d.AddBy2Points(x1:=0, y1:=0.15, x2:=0, y2:=0)
            objArc2 = objBaseProfile(0).Arcs2d.AddByCenterStartEnd(xCenter:=0.025, yCenter:=0, _
                                                                       xStart:=0, yStart:=0, xEnd:=0.05, yEnd:=0)
            objRelations = objBaseProfile(0).Relations2d
            objRelations.AddKeypoint(Object1:=objLine1, Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objArc1, Index2:=SolidEdgeConstants.KeypointIndexConstants.igArcStart)
            objRelations.AddKeypoint(Object1:=objArc1, Index1:=SolidEdgeConstants.KeypointIndexConstants.igArcEnd, Object2:=objLine2, Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            objRelations.AddKeypoint(Object1:=objLine2, Index1:=SolidEdgeConstants.KeypointIndexConstants.igLineEnd, Object2:=objArc2, Index2:=SolidEdgeConstants.KeypointIndexConstants.igArcStart)
            objRelations.AddKeypoint(Object1:=objArc2, Index1:=SolidEdgeConstants.KeypointIndexConstants.igArcEnd, Object2:=objLine1, Index2:=SolidEdgeConstants.KeypointIndexConstants.igLineStart)
            lngStatus = objBaseProfile(0).End(ValidationCriteria:=SolidEdgeConstants.ProfileValidationType.igProfileClosed)
            If lngStatus <> 0 Then
                MsgBox("Base feature profile is not closed")
            End If

            objModel = objDoc.Models.AddFiniteExtrudedProtrusion(NumberOfProfiles:=1, ProfileArray:=objBaseProfile, _
                                                                     ProfilePlaneSide:=SolidEdgePart.FeaturePropertyConstants.igRight, ExtrusionDistance:=0.05)
            objBaseProfile(0).Visible = False
            objExtProtrusion = objModel.ExtrudedProtrusions(0)
            If objExtProtrusion.Status <> SolidEdgePart.FeatureStatusConstants.igFeatureOK Then
                MsgBox("Base Feature is not created properly")
            End If
            'Create SweptProtrusion with Add method
            objTraceRefPlane = objDoc.RefPlanes.AddParallelByDistance(ParentPlane:=objDoc.RefPlanes(1), _
                                                                          Distance:=0.05, NormalSide:=SolidEdgePart.FeaturePropertyConstants.igRight, Local:=True)
            objTraceprofile(0) = objDoc.ProfileSets.Add.Profiles.Add(pRefPlaneDisp:=objTraceRefPlane)
            objLine1 = objTraceprofile(0).Lines2d.AddBy2Points(x1:=0.05, y1:=0, x2:=0.05, y2:=0.15)
