Remarks

This API will not work with Bolt or Edge type connectors.

For Bolt type of connector use "GetBoltConnectionOptions" for Edge type of connector use "GetEdgeConnectorOptions"

using System.Runtime.InteropServices;

internal static class Module1
{

    public static void Main()
    {
        SolidEdgeFramework.Application objApplication = null;
        SolidEdgeAssembly.AssemblyDocument objAsmDocument = null;
        SolidEdgePart.StudyOwner objStudyOwner = null;
        SolidEdgePart.Study objStudy = null;
        SolidEdgePart.ConnectorOwner objConnectorOwner = null;
        SolidEdgePart.FEAConnector objConnector = null;
        double dMinSearchDistance = 0.1;

        double dConneterVal = 0; // Connector Value
        double dSearchDistance = 0; // Search distance
        double dMinSearchDist = 0; // Minimum search Distance
        double dPenaltyVal = 0; // Penalty value
        double xx = 0; // GPosition X – cord
        double yy = 0; // GPosition Y – cord
        double zz = 0; // GPosition Z – cord
        uint uiColorConnectorSymbol = 0; // Color of connector symbol
        double dSpacingFactor = 0; // Symbol spacing factor
        double dSizeFactor = 0; // Symbol size factor
        uint uiConnProps = 0; // Connector properties
        double dConnectorSymSpacing = 0; // Connector Symbol Spacing
        double dConnSymSpacSize = 0; // Connector symbol size

        try
        {

            // Connect to Solid Edge
            objApplication = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
            objAsmDocument = objApplication.ActiveDocument;

            objStudyOwner = objAsmDocument.StudyOwner;
            objStudy = objStudyOwner.Item(1);
            objStudy.GetConnectorOwner(objConnectorOwner);
            objConnector = objConnectorOwner.Item(1);

            objConnector.GetConnectorOptions(dConneterVal, dSearchDistance, dMinSearchDist, dPenaltyVal, xx, yy, zz, uiColorConnectorSymbol, dSpacingFactor, dSizeFactor, uiConnProps, dConnectorSymSpacing, dConnSymSpacSize, SolidEdgePart.FEAInitialPenetrationTypeEnum_Auto.eCalculatedType_Auto);



        }
        catch (Exception ex)
        {
            Console.WriteLine(ex.Message);
        }
        finally
        {
            objApplication = null;
            objAsmDocument = null;
            objStudy = null;
            objStudyOwner = null;
            objConnector = null;
        }

    }

}
