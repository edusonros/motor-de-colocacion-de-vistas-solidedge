using System;
using System.Runtime.InteropServices;

namespace Examples
{
    class Program
    {
        [STAThread]
        static void Main(string[] args)
        {
            SolidEdgeFramework.Application application = null;
            SolidEdgeAssembly.AssemblyDocument assemblyDocument = null;
            SolidEdgeAssembly.PhysicalProperties physicalProperties = null;

            try
            {
                // See "Handling 'Application is Busy' and 'Call was Rejected By Callee' errors" topic.
                OleMessageFilter.Register();

                // Attempt to connect to a running instance of Solid Edge.
                application = (SolidEdgeFramework.Application)Marshal.GetActiveObject("SolidEdge.Application");
                assemblyDocument = application.ActiveDocument as SolidEdgeAssembly.AssemblyDocument;

                if (assemblyDocument != null)
                {
                    physicalProperties = assemblyDocument.PhysicalProperties;

                    physicalProperties.UserDefinedPropertiesStatus = true;

                    double Volume = 0.135;
                    double Area = 57;
                    double Mass = 5;
                    var CenterOfMass = (System.Array)new double[] { 15, 0.5, 5 };
                    var CenterOfVolume = (System.Array)new double[] { 35, 0.05, 53 };
                    var GlobalMoments = (System.Array)new double[] { 1, 4, 3, 5, 8, 7, 9 };
                    var PrincipalMoments = (System.Array)new double[] { 4, 3, 5 };
                    var PrincipalAxis1 = (System.Array)new double[] { 3, 9, 5 };
                    var PrincipalAxis2 = (System.Array)new double[] { 7, 5, 6 };
                    var PrincipalAxis3 = (System.Array)new double[] { 5, 8, 5 };
                    var RadiiOfGyration = (System.Array)new double[] { 1, 5, 2 };
                    double OverrideMass = 15;

                    physicalProperties.SetUserDefinedAssemblyPhysicalProperties(
                        ref Volume,
                        ref Area,
                        ref Mass,
                        ref CenterOfMass,
                        ref CenterOfVolume,
                        ref GlobalMoments,
                        ref PrincipalMoments,
                        ref PrincipalAxis1,
                        ref PrincipalAxis2,
                        ref PrincipalAxis3,
                        ref RadiiOfGyration,
                        ref OverrideMass);
                }
            }
            catch (System.Exception ex)
            {
                Console.WriteLine(ex);
            }
            finally
            {
                OleMessageFilter.Unregister();
            }
        }
    }
}
